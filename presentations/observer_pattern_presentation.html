    PRICE_CHANGE = "price_change"

class Product:
    """◊û◊ï◊¶◊® ◊ë◊ó◊†◊ï◊™ - ◊ñ◊î ◊î-Subject"""
    def __init__(self, name, price, stock):
        self.name = name
        self.price = price
        self.stock = stock
        self._observers = []

    def attach(self, observer):
        self._observers.append(observer)

    def detach(self, observer):
        self._observers.remove(observer)

    def notify(self, notification_type, data):
        for observer in self._observers:
            observer.update(notification_type, data)

    def set_price(self, new_price):
        """◊©◊ô◊†◊ï◊ô ◊û◊ó◊ô◊®"""
        old_price = self.price
        self.price = new_price

        if new_price != old_price:
            self.notify(NotificationType.PRICE_CHANGE, {
                'product': self.name,
                'old_price': old_price,
                'new_price': new_price
            })

    def sell(self, quantity):
        """◊û◊õ◊ô◊®◊™ ◊û◊ï◊¶◊®"""
        if self.stock >= quantity:
            self.stock -= quantity

            # ◊î◊™◊®◊ê◊î ◊¢◊ú ◊û◊ú◊ê◊ô ◊†◊û◊ï◊ö
            if self.stock < 5:
                self.notify(NotificationType.STOCK_LOW, {
                    'product': self.name,
                    'remaining_stock': self.stock
                })
