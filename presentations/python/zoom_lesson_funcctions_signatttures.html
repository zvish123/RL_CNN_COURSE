פרמטרים מפורשים ⭐

positional arguments 
def get_words_with_unknown(cipher_text, cracked_text):
    # Python מוודא שמעבירים בדיוק 2 פרמטרים!
    return [(c,r) for c,r in zip(cipher_text.split(), cracked_text.split())]

# הסדר חשוב! cipher קודם, אחר כך cracked
words_info = get_words_with_unknown(cipher_text, cracked_text)
keyword arguments 
def get_words_with_unknown(cipher_text, cracked_text):
    return [(c,r) for c,r in zip(cipher_text.split(), cracked_text.split())]

# עם keyword arguments — הסדר לא משנה!
words_info = get_words_with_unknown(
    cracked_text=cracked_text,
    cipher_text=cipher_text
)

*args — פרמטרים משתנים

*args magic 
def get_words_with_unknown(*params):
    # params הוא tuple — Python ארז הכל שם
    if len(params) == 1:
        cracked_text = params[0]
        cipher_text  = None
    elif len(params) == 2:
        cipher_text, cracked_text = params[0], params[1]
    else:
        raise ValueError("יותר מדי פרמטרים!")
    return [(c,r) for c,r in zip(cipher_text.split(), cracked_text.split())]

# שתי הקריאות עובדות!
words_info = get_words_with_unknown(cracked_text)
words_info = get_words_with_unknown(cipher_text, cracked_text)

def decode_caesar(*params):
    text  = params[0]
    shift = params[1]
    return ''.join(
        chr((ord(c)-shift-65)%26+65)
        for c in text
    )
בונוס: הוסיפו ערך ברירת מחדל ל-shift
def decode_caesar(text, shift=3):
    return ''.join(
        chr((ord(c)-shift-65)%26+65)
        for c in text
    )
decode_caesar("KHOOR")     # shift=3
decode_caesar("KHOOR", 5)  # shift=5

def send_alert(message, level="info", timestamp=None):
    prefix = {"info":"[INFO]", "warning":"[WARN]",
               "error":"[ERR] "}.get(level, "[INFO]")
    ts = f" ({timestamp})" if timestamp else ""
    print(f"{prefix} {message}{ts}")

send_alert("נותק", level="error")

💬 שאלות להבנה — לחצו על הכפתורים בצ'אט: מי יכול לענות לפני שמגלים?

▶
מה קורה אם מעבירים 3 פרמטרים לפונקציה שמצפה ל-2?
Python יזרוק: TypeError: takes 2 positional arguments but 3 were given. זה יתרון — שגיאה מיידית, לא בריצה!
▶
מה ההבדל בין f(a, b) לבין f(b=b, a=a)?
בראשונה — פוזיציונלי, הסדר קובע. בשנייה — keyword arguments, הסדר לא משנה. שתיהן עובדות עם אותה הגדרה!
▶
למה *args פחות מומלץ אף שהוא הכי גמיש?
כי מי שקורא f(*params) לא יודע מה להעביר — חייב לקרוא את הקוד פנימה. גם שגיאות יופיעו רק ב-runtime.
