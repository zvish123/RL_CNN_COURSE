def get_word_pattern(word):
    word = word.upper()
    next_num = 0
    letter_numbers_mapping = {}
    word_pattern = []
    
    for letter in word:
        if letter not in letter_numbers_mapping:
            letter_numbers_mapping[letter] = str(next_num)
            next_num += 1
        word_pattern.append(letter_numbers_mapping[letter])
    
    return '.'.join(word_pattern)
