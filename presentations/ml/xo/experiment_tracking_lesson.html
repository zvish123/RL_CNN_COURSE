            with open(self.csv_file, 'w', newline='', encoding='utf-8') as f:

                writer.writerow([
                    'Timestamp', 'Method', 'Gamma', 'Epsilon',
                    'WinRate', 'DictSize', 'TrainingTime', 'Notes'

                params.get('gamma', ''),
                params.get('epsilon', ''),
                results.get('win_rate', ''),
                results.get('dict_size', ''),
                results.get('training_time', ''),
                notes
