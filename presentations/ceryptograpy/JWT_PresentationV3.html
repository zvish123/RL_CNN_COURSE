01 — הגדרה

מה זה JWT?

JWT הוא תקן מבוסס JSON להעברת מידע מאובטח בין צדדים. המידע חתום דיגיטלית — ניתן לאמת שלא שונה.

סיכום זרימה כללית

1
לקוח
שולח credentials
alice:password
2
שרת
מאמת + יוצר JWT
jwt.encode()
3
לקוח
מקבל ושומר JWT
token = recv()
4
לקוח
שולח JWT + פעולה
send(token|action)
5
שרת
מאמת + בודק הרשאה
jwt.decode()
עקרון הזהב: SECRET_KEY נמצא אך ורק בשרת — הלקוח מקבל טוקן חתום, לעולם לא רואה את המפתח ולעולם לא יוצר טוקן בעצמו.
02 — מבנה

מבנה JWT

JWT מורכב משלושה חלקים מופרדים בנקודות:

Header (כותרת)
סוג האלגוריתם לחתימה
{
  "alg": "HS256",
  "typ": "JWT"
}
Payload (מטען)
נתוני המשתמש והרשאות
{
  "user_id": "alice",
  "role": "admin",
  "exp": 1712345678
}
Signature (חתימה)
מונעת זיוף — נוצרת עם SECRET_KEY
HMACSHA256(
  header+"."+payload,
  SECRET_KEY
)

03 — דוגמת קוד

שרת — socket + JWT

זרימה סינכרונית עם socket + threading לריבוי לקוחות:

// קוד שרת

import socket, jwt, os, threading
from datetime import datetime, timedelta

# SECRET_KEY נמצא רק בשרת — נטען ממשתני סביבה
SECRET_KEY = os.getenv("SECRET_KEY", "mysecret")
USERS_DB = { "alice": "1234", "bob": "5678" }

def create_token(username):
    payload = {"user_id": username,
               "exp": datetime.utcnow() + timedelta(hours=1)}
    return jwt.encode(payload, SECRET_KEY, algorithm="HS256")

def verify_token(token):
    try:
        return jwt.decode(token, SECRET_KEY, algorithms=["HS256"])
    except (jwt.ExpiredSignatureError, jwt.InvalidTokenError):
        return None

def handle_client(conn, addr):
    with conn:
        # שלב 1: קבלת credentials
        username, password = conn.recv(1024).decode().split(":", 1)
        if USERS_DB.get(username) != password:
            conn.sendall(b"ERROR: bad credentials"); return
        # שלב 2: יצירת JWT ושליחה
        token = create_token(username)
        conn.sendall(token.encode())
        # שלב 3: קבלת בקשה עם הטוקן
        token_recv = conn.recv(4096).decode()
        payload = verify_token(token_recv)
        msg = f"אומת: {payload['user_id']}" if payload else "ERROR"
        conn.sendall(msg.encode())

def main():
    srv = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    srv.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    srv.bind(("localhost", 9000)); srv.listen(5)
    while True:
        conn, addr = srv.accept()
        threading.Thread(target=handle_client, args=(conn, addr)).start()

main()
04 — דוגמת קוד

לקוח — socket

⚠️ חשוב: הלקוח לא מכיר את SECRET_KEY בכלל — רק שולח credentials ומקבל טוקן מהשרת.

// קוד לקוח

import socket

def communicate(username, password):
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
        s.connect(("localhost", 9000))

        # שלב 1: שליחת פרטי התחברות
        s.sendall(f"{username}:{password}".encode())

        # שלב 2: קבלת JWT מהשרת
        token = s.recv(4096).decode()
        if token.startswith("ERROR"):
            print(token); return
        print(f"קיבלנו טוקן: {token[:40]}...")

        # שלב 3: שימוש בטוקן לבקשות הבאות
        s.sendall(token.encode())
        response = s.recv(1024).decode()
        print(f"תגובה: {response}")

communicate("alice", "1234")

// סיכום זרימה

שרת                              לקוח
|── alice:1234 (TCP) ──────────→ |  אימות מול USERS_DB
|← JWT (bytes, תוקף שעה) ────────|  jwt.encode() + SECRET_KEY
|── JWT ───────────────────────→ |  jwt.decode() + אימות
|← אומת בהצלחה: alice ────────── |
     [ SECRET_KEY נמצא אך ורק בשרת ]

06 — הרשאות

ארבעת סוגי ההרשאות ב-JWT

ניתן לשלב מספר שיטות יחד ב-Payload אחד:

סוגשדה ב-JWTערך לדוגמהבדיקה בשרת
RBAC role "admin" רמת תפקיד ≥ נדרש
Permission permissions ["read","delete"] action in list
Scope scope "read:all write:all" scope in split()
Resource resources {"reports":"admin"} רמת משאב ≥ נדרש
alice (admin) role:"admin" permissions:["read","write","delete"] scope:"read:all write:all"
bob (viewer) role:"viewer" permissions:["read"] scope:"read:reports"
07 — קוד

פונקציות בדיקת הרשאה

# 1. RBAC — תפקיד מספיק?
def check_rbac(payload, required_role):
    level = {"admin":3, "editor":2, "viewer":1}
    return level.get(payload["role"],0) >= level.get(required_role,99)

# 2. Permission-Based — הפעולה ברשימה?
def check_permission(payload, action):
    return action in payload.get("permissions", [])

# 3. Scope-Based — ה-scope כולל את הדרוש?
def check_scope(payload, required_scope):
    return required_scope in payload.get("scope","").split()

# 4. Resource-Based — הרשאה למשאב ספציפי?
def check_resource(payload, resource, action):
    level = {"admin":3, "write":2, "read":1}
    user_perm = payload.get("resources",{}).get(resource,"")
    return level.get(user_perm,0) >= level.get(action,99)
08 — קוד מלא

שרת מלא — socket + הרשאות

פורמט בקשה: TOKEN|TYPE|VALUE

def handle_client(conn, addr):
    with conn:
        # Login
        username, password = conn.recv(1024).decode().split(":",1)
        if USERS_DB.get(username,{}).get("password") != password:
            conn.sendall(b"ERROR"); return
        conn.sendall(create_token(username).encode())

        while True:
            raw = conn.recv(4096).decode().strip()
            if not raw: break
            token_recv, check_type, value = raw.split("|")
            payload = jwt.decode(token_recv, SECRET_KEY,
                                 algorithms=["HS256"])

            if   check_type == "rbac":
                ok = check_rbac(payload, value)
            elif check_type == "permission":
                ok = check_permission(payload, value)
            elif check_type == "scope":
                ok = check_scope(payload, value)
            elif check_type.startswith("resource:"):
                resource = check_type.split(":",1)[1]
                ok = check_resource(payload, resource, value)

            conn.sendall(("OK" if ok else "DENIED").encode())

10 — סיכום

יתרונות וחסרונות של JWT

יתרונות
החתימה מוודאת שהאסימון לא שונה
נישא עצמאית — כל המידע בטוקן, ללא שאילתת DB
גמיש — תומך RBAC, Permissions, Scope, Resource
ניתן לשלוח בכל חיבור: HTTP, socket
תוקף מובנה (exp) — פג אוטומטית
חסרונות
אם נגנב — אי אפשר לבטל לפני פקיעת תוקף
דורש ניהול blacklist לביטול מוקדם
SECRET_KEY שנחשף = כל הטוקנים בסכנה
Payload גדול → בקשות כבדות יותר
חייבים לשמור SECRET_KEY בצורה מאובטחת
