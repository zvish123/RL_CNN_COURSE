💡 הפתרון: מחלקות (Classes)!

מחלקה היא כמו "תבנית" ליצירת אובייקטים

class Student:
    def __init__(self, name, age):
        self.name = name
        self.age = age
        self.grades = []
    
    def add_grade(self, grade):
        self.grades.append(grade)
    
    def calculate_average(self):
        if self.grades:
            return sum(self.grades) / len(self.grades)
        return 0

# עכשיו נוכל ליצור כמה תלמידים שנרצה!
student1 = Student("דני", 20)
student2 = Student("מיכל", 19)

רכיב מה זה? דוגמה
__init__ הבנאי - פונקציה שרצה כשיוצרים אובייקט חדש def __init__(self, name)
self הפניה לאובייקט הנוכחי self.name = name
שדות (Fields) משתנים השייכים לאובייקט self.name, self.age
מתודות (Methods) פונקציות השייכות למחלקה def calculate_average(self)

🏦 דוגמה מעשית: חשבון בנק

class BankAccount:
    def __init__(self, owner_name, initial_balance=0):
        self.owner_name = owner_name
        self.balance = initial_balance
        self.transaction_history = []
    
    def deposit(self, amount):
        if amount > 0:
            self.balance += amount
            self.transaction_history.append(f"הפקדה: +{amount}")
            return f"הופקדו {amount} ש״ח. יתרה נוכחית: {self.balance}"
        return "סכום לא חוקי"
    
    def withdraw(self, amount):
        if amount > 0 and amount <= self.balance:
            self.balance -= amount
            self.transaction_history.append(f"משיכה: -{amount}")
            return f"נמשכו {amount} ש״ח. יתרה נוכחית: {self.balance}"
        return "לא ניתן למשוך - יתרה לא מספיקה או סכום לא חוקי"
    
    def get_balance(self):
        return f"יתרה עבור {self.owner_name}: {self.balance} ש״ח"

✨ מתודות מיוחדות (Magic Methods)

class Student:
    def __init__(self, name, age):
        self.name = name
        self.age = age
        self.grades = []
    
    def __str__(self):
        return f"תלמיד: {self.name}, גיל: {self.age}"
    
    def __repr__(self):
        return f"Student('{self.name}', {self.age})"
    
    def __len__(self):
        return len(self.grades)
    
    def __add__(self, other):
        # איחוד של שני תלמידים ליצירת קבוצה
        return f"קבוצה: {self.name} ו-{other.name}"
