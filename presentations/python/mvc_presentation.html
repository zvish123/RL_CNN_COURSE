Model â€” mvc_model.py

import json

class UserModel:
    def __init__(self):
        self.users = []

    def load_users_from_json(self, filename):
        with open(filename, 'r') as f:
            self.users = json.load(f)

    def get_users(self):
        return self.users

    def save_users_to_json(self, filename, users):
        with open(filename, 'w') as f:
            json.dump(users, f, indent=2)

    def greet_user(self, row):
        try:
            user = self.users[row]
            return f"Hello {user['username']}!"
        except IndexError:
            return "No user selected"

from PyQt6.QtWidgets import (
    QWidget, QVBoxLayout,
    QHBoxLayout, QTableWidget,
    QTableWidgetItem, QPushButton
)

class UserView(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Users")
        layout = QVBoxLayout(self)

        # Table
        self.table = QTableWidget(self)
        self.table.setColumnCount(2)
        self.table.setHorizontalHeaderLabels(
            ["Username", "Email"])
        layout.addWidget(self.table)

        # Buttons
        btn_row = QHBoxLayout()
        self.selectBtn = QPushButton("Select")
        self.saveBtn   = QPushButton("Save")
        btn_row.addWidget(self.selectBtn)
        btn_row.addWidget(self.saveBtn)
        layout.addLayout(btn_row)

    def update_view(self, users):
        self.table.setRowCount(len(users))
        for row, user in enumerate(users):
            self.table.setItem(
                row, 0,
                QTableWidgetItem(user['username']))
            self.table.setItem(
                row, 1,
                QTableWidgetItem(user['email']))

import json
from mvc_model import UserModel
from mvc_view  import UserView

USERS_FILE = "users.json"

class UserController:
    def __init__(self):
        self.model = UserModel()
        self.view  = UserView()
        # Connect signals â†’ slots
        self.view.selectBtn.clicked.connect(
            self.select)
        self.view.saveBtn.clicked.connect(
            self.save)

    def run(self):
        self.model.load_users_from_json(USERS_FILE)
        self.view.update_view(
            self.model.get_users())
        self.view.show()

    def select(self):
        row = self.view.table.currentRow()
        msg = self.model.greet_user(row)
        print(msg)

    def save(self):
        users = []
        for r in range(
                self.view.table.rowCount()):
            users.append({
                "username": self.view.table
                    .item(r,0).text(),
                "email": self.view.table
                    .item(r,1).text()
            })
        self.model.save_users_to_json(
            USERS_FILE, users)
