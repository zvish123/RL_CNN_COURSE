X = df.iloc[:, :-1].values  # כל העמודות חוץ מהאחרונה

# שימוש ב-SimpleImputer להשלמת ערכים חסרים
from sklearn.impute import SimpleImputer

imputer = SimpleImputer(missing_values=np.nan, strategy='mean')
imputer = imputer.fit(X[:, 1:3])  # רק עמודות Age ו-Albumin
X[:, 1:3] = imputer.transform(X[:, 1:3])

משתנים עם טווחי ערכים שונים יכולים להטות את האלגוריתם. למשל: גיל (20-80) מול רמת חלבון (3-4)

🤖 הרצת מודל למידת מכונה

🧠 MLPClassifier - מה זה?

Multi-Layer Perceptron - רשת נוירונים פשוטה שלומדת לסווג נתונים

import sklearn.neural_network

# הגדרת המודל
mlp = sklearn.neural_network.MLPClassifier(
    hidden_layer_sizes=(100,),    # שכבה נסתרת עם 100 נוירונים
    max_iter=200,                 # מספר איטרציות מקסימלי
    solver='sgd',                  # אלגוריתם האופטימיזציה
    learning_rate_init=0.05       # קצב הלמידה
)

# אימון המודל
mlp.fit(X_train, Y_train)

# בדיקת הביצועים
training_score = mlp.score(X_train, Y_train)
test_score = mlp.score(X_test, Y_test)
