class Dog:
    def __init__(self, name="Blacky", age=0):
        self.name = name
        self.age = age

        self.age += 1
    
    def get_age(self):

  • self - מצביע על האובייקט הנוכחי
  • methods - הפונקציות שלמחלקה
  • attributes - המשתנים של האובייקט

🔍 הפרמטר self

❓ למה אנחנו צריכים את self?

  • איך פייתון יודע על איזה אובייקט להפעיל את המתודה?
  • הרי יכולים להיות הרבה instances של אותו class!

💡 הפתרון - self:

  • self מצביע על האובייקט שעליו המתודה פועלת
  • לכן מוסיפים למתודות את הפרמטר self
  • פייתון מעביר אותו אוטומטית!
class Dog:

איך נמנע שינויים לא רצויים?

🛡️ הפתרון - משתנים פרטיים:

class Dog:
    def __init__(self, name, age=0):
        self.__name = name    # __ = פרטי
        self.__age = age      # __ = פרטי
    
    def get_age(self):        # accessor
        return self.__age
    
    def set_age(self, age):   # mutator
        if age >= 0:          # בדיקת תקינות
            self.__age = age
        else:
            print("גיל חייב להיות חיובי!")
    
    def get_name(self):
        return self.__name

# כעת לא ניתן לגשת ישירות
dog = Dog("רקס", 3)
# dog.__age = -5    # לא יעבוד!
dog.set_age(4)      # הדרך הבטוחה

👨‍👩‍👧‍👦 ירושה - Inheritance

🤔 מה זה ירושה?

  • לעיתים class הוא סוג ספציפי של class אחר
  • חתול הוא חיה, מכונית היא כלי תחבורה
  • במקום לכתוב הכל מחדש - נירש מהקיים!

🏗️ מבנה ירושה:

Animal (מחלקת אב)
⬇️
Dog
Cat

Dog ו-Cat יורשים מ-Animal

🐾 דוגמה לירושה

# מחלקת האב - Animal
class Animal:
    def __init__(self, name, age=0):
        self.name = name
        self.age = age
    
    def eat(self):
        print(f"{self.name} אוכל")
    
    def sleep(self):
        print(f"{self.name} ישן")
    
    def __str__(self):
        return f"{self.name} (גיל {self.age})"

# מחלקת הבן - Dog יורש מ-Animal
class Dog(Animal):
    def __init__(self, name, age=0, breed="מעורב"):
        super().__init__(name, age)  # קריאה למחלקת האב
        self.breed = breed
    
    def bark(self):              # מתודה ייחודית לכלב
        print(f"{self.name}: הב הב!")
    
    def __str__(self):
        return f"🐕 כלב {super().__str__()}, גזע: {self.breed}"

# מחלקת הבן - Cat יורש מ-Animal  
class Cat(Animal):
    def meow(self):              # מתודה ייחודית לחתול
        print(f"{self.name}: מיאו!")
    
    def __str__(self):
        return f"🐱 חתול {super().__str__()}"

🎯 שימוש בירושה

🚀 בואו נשתמש במחלקות:

# יצירת חיות
my_dog = Dog("רקס", 3, "לברדור")
my_cat = Cat("מיטב", 2)

print("🐾 החיות שלנו:")

print(my_cat)    # 🐱 חתול מיטב (גיל 2)

print("\n🍽️ זמן אוכל:")
my_dog.eat()     # רקס אוכל (ירושה מ-Animal)
my_cat.eat()     # מיטב אוכל (ירושה מ-Animal)

print("\n🔊 קולות:")
my_dog.bark()    # רקס: הב הב! (ייחודי לכלב)
my_cat.meow()    # מיטב: מיאו! (ייחודי לחתול)

print("\n😴 זמן שינה:")
my_dog.sleep()   # רקס ישן (ירושה מ-Animal)
my_cat.sleep()   # מיטב ישן (ירושה מ-Animal)

  • קוד ניתן לשימוש חוזר - לא כותבים הכל מחדש
  • הירארכיה ברורה - קל להבין את הקשרים
  • קל להוסיף תכונות חדשות למחלקות בנות

🔄 דריסת מתודות - Method Overriding

🎭 מה זה דריסת מתודות?

לפעמים רוצים שמחלקת הבן תתנהג אחרת ממחלקת האב

🐕 דוגמה - כלב רועה מיוחד:

class Animal:
    def make_sound(self):
        print("החיה משמיעה קול")

class Dog(Animal):
    def make_sound(self):        # דריסה של המתודה
        print("הכלב נובח: הב הב!")

class ShepherdDog(Dog):
    def make_sound(self):        # דריסה נוספת
        super().make_sound()     # קריאה למתודה של האב
        print("וגם משמיע קול התרעה!")

# שימוש
regular_animal = Animal()
my_dog = Dog()
shepherd = ShepherdDog()

regular_animal.make_sound()  # החיה משמיעה קול
my_dog.make_sound()          # הכלב נובח: הב הב!
shepherd.make_sound()        # הכלב נובח: הב הב!
                            # וגם משמיע קול התרעה!

🔍 בדיקת סוג האובייקט - isinstance

❓ איך נדע איזה סוג אובייקט יש לנו?

לפעמים יש לנו רשימה של חיות מעורבות ורוצים לבדוק מי מהן כלב

🎯 הפונקציה isinstance:

# יצירת חיות שונות
animals = [
    Dog("רקס", 3),
    Cat("מיטב", 2),
    Dog("מקס", 5),
    Animal("חיה כללית", 1)
]

print("🔍 בדיקת סוגי החיות:")
for animal in animals:
    print(f"\n{animal.name}:")
    
    if isinstance(animal, Dog):
        print("  ✅ זה כלב!")
        animal.bark()
    
    if isinstance(animal, Cat):
        print("  ✅ זה חתול!")
        animal.meow()
    
    if isinstance(animal, Animal):
        print("  ✅ זה חיה (כולל כלבים וחתולים)")

# תוצאה לדוגמה:
# רקס:
#   ✅ זה כלב!
#   רקס: הב הב!
#   ✅ זה חיה (כולל כלבים וחתולים)

💡 חשוב לזכור:

  • אובייקט תמיד שייך גם למחלקת האב שלו
  • Dog isinstance של Animal מחזיר True

class Person:
    def __init__(self, name, age):
        self.__name = name
        self.__age = age
    
    def get_name(self):
        return self.__name
    
    def get_age(self):
        return self.__age
    
    def set_age(self, age):
        if age >= 0:
            self.__age = age
    
    def __str__(self):
        return f"{self.__name} (גיל {self.__age})"

class Teacher(Person):
    def __init__(self, name, age, subject, salary):
        super().__init__(name, age)
        self.__subject = subject
        self.__salary = salary
    
    def teach(self):
        return f"{self.get_name()} מלמד {self.__subject}"
    
    def get_subject(self):
        return self.__subject
    
    def __str__(self):
        return f"👨‍🏫 מורה {super().__str__()}, מקצוע: {self.__subject}"

class Student(Person):
    def __init__(self, name, age, grade_level, average):
        super().__init__(name, age)
        self.__grade_level = grade_level
        self.__average = average
    
    def study(self):
        return f"{self.get_name()} לומד בכיתה {self.__grade_level}"
    
    def get_average(self):
        return self.__average
    
    def set_average(self, average):
        if 0 <= average <= 100:
            self.__average = average
    
    def __str__(self):
        return f"👨‍🎓 תלמיד {super().__str__()}, כיתה {self.__grade_level}, ממוצע: {self.__average}"

    def get_subject(self):
        return self.__subject

    def __str__(self):
        return f"👨‍🎓 תלמיד {super().__str__()}, כיתה {self.__grade_level}, ממוצע: {self.__average}"
}"

✅ פתרון תרגיל 2

class Pet:
    def __init__(self, name, age, price):
        self.__name = name
        self.__age = age
        self.__price = price
    
    def get_name(self):
        return self.__name
    
    def get_price(self):
        return self.__price
    
    def make_sound(self):
        return f"{self.__name} משמיע קול"
    
    def __str__(self):
        return f"{self.__name} (גיל {self.__age}, מחיר: {self.__price}₪)"

class Dog(Pet):
    def make_sound(self):
        return f"🐕 {self.get_name()}: הב הב!"

class Cat(Pet):
    def make_sound(self):
        return f"🐱 {self.get_name()}: מיאו!"

class Bird(Pet):
    def make_sound(self):
        return f"🐦 {self.get_name()}: צ'יפ צ'יפ!"

class PetStore:
    def __init__(self, name):
        self.name = name
        self.pets = []
    
    def add_pet(self, pet):
        self.pets.append(pet)
        print(f"✅ {pet.get_name()} נוסף לחנות")
    
    def show_all_pets(self):
        print(f"\n🏪 {self.name} - רשימת חיות:")
        for i, pet in enumerate(self.pets, 1):
            print(f"  {i}. {pet}")
    
    def pets_concert(self):
        print(f"\n🎵 קונצרט בחנות {self.name}:")
        for pet in self.pets:
            print(f"  {pet.make_sound()}")
    
    def total_value(self):
        return sum(pet.get_price() for pet in self.pets)
    
    def find_pets_by_type(self, pet_type):
        return [pet for pet in self.pets if isinstance(pet, pet_type)]
