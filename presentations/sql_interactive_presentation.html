    s.name AS student_name,
    COUNT(e.course_id) AS num_courses,
    AVG(e.grade) AS avg_grade,
    MAX(e.grade) AS highest_grade
FROM students s
INNER JOIN enrollments e ON s.student_id = e.student_id
