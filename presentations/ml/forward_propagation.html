def feed_forward(X, W1, b1, W2, b2):
    # שכבה 1
    A0 = X.T             # (n × m)
    Z1 = W1 @ A0 + b1   # Z⁽¹⁾
    A1 = sigmoid(Z1)    # A⁽¹⁾
    # שכבה 2
    Z2 = W2 @ A1 + b2   # Z⁽²⁾
    A2 = sigmoid(Z2)    # ŷ
    return A2
