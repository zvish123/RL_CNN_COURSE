מה זה Word Pattern?

כל מילה מקבלת pattern שמבוסס על מיקום האותיות החוזרות. מכיוון שההצפנה מחליפה תו בתו, ה-pattern נשמר אחרי הצפנה!

← אותו pattern!
✅ XYZZY מוצפן ↔ יכול להיות HELLO, PUPPY, BELLE וכל מילה עם pattern 0.1.2.2.3

כיצד מחשבים pattern?
def get_word_pattern(word: str) -> str:
    next_num = 0
    char_map = {}
    pattern  = []
    for ch in word.upper():
        if ch not in char_map:
            char_map[ch] = str(next_num)
            next_num += 1
        pattern.append(char_map[ch])
    return '.'.join(pattern)

# דוגמאות:
get_word_pattern("HELLO")      # → "0.1.2.2.3"
get_word_pattern("SIMPLE")     # → "0.1.2.3.4.5"
get_word_pattern("RECONSIDER") # → "0.1.2.3.4.5.6.1.0.7"

פונקציית ה-Intersect

החיתוך הוא הלב של האלגוריתם — ממזג מיפויים משתי מילים שונות:

def intersect_mapping(map_a, map_b) -> dict:
    result = get_blank_mapping()
    for ch in ALPHABET:
        a_list = map_a.get(ch, [])
        b_list = map_b.get(ch, [])

        if not a_list:                # מפה א אינה מגבילה
            result[ch] = b_list
        elif not b_list:              # מפה ב אינה מגבילה
            result[ch] = a_list
        else:                         # חיתוך!
            result[ch] = [x for x in a_list if x in b_list]
    return result

מה זה Complete Mapping?

לאחר שלב Word Patterns, חלק מהאותיות עדיין לא נפתרו. Complete Mapping ממשיך:

מחפשים מילים שרוב האותיות בהן פוענחו, ונשאר ? יחיד. אז בודקים מה יכולה להיות אותה אות.

✅ דוגמה: אם פוצחנו "HI??EN" — רק "HIDDEN" עובדת. לכן D→D!

def try_complete_single_unknown(cipher_word, cracked_word, mapping):
    # מוצא את מיקום ה-? היחיד
    pos = cracked_word.index('?')
    cipher_char = cipher_word[pos]     # תו cipher שצריך מיפוי
    unused = get_unused_letters(mapping)  # אותיות שעוד לא ממופות

    # בנה regex מהמילה החלקית
    pattern_parts = []
    for ch in cracked_word:
        pattern_parts.append('[A-Z]' if ch == '?'
                              else re.escape(ch.upper()))
    word_regex = re.compile('.'.join(pattern_parts), re.IGNORECASE)

    valid_options = []
    for candidate in unused:
        if not ('A' <= candidate.upper() <= 'Z'): continue
        test_word = (cracked_word[:pos] +
                     candidate.upper() +
                     cracked_word[pos+1:]).upper()
        if word_regex.match(test_word):
            pat = get_word_pattern(test_word)
            if pat in all_patterns and test_word in all_patterns[pat]:
                valid_options.append((cipher_char, candidate.upper()))

    # רק פתרון יחיד מקובל!
    return valid_options[0] if len(valid_options) == 1 else None

def complete_mapping(cipher_text, cracked_text, mapping, space_char):
    for round_num in range(1, N + 1):
        unmapped = get_unmapped_keys(mapping)
        if not unmapped:
            break  # כל התווים מופו!

        words_info = get_words_with_unknowns(
            cipher_text, cracked_text, space_char)  # ממוין לפי ? עולה

        found = False
        for cipher_word, cracked_word, q_count in words_info:
            if q_count != 1: continue  # רק ? יחיד

            result = try_complete_single_unknown(
                cipher_word, cracked_word, mapping)

            if result:
                c_ch, p_ch = result
                mapping[c_ch] = [p_ch]      # עדכן מיפוי
                mapping = remove_solved(mapping)
                cracked_text = translate(mapping, cipher_text)
                found = True
                break  # סיבוב חדש!

        if not found: break  # אין התקדמות

    return cracked_text, mapping

def translate_message(message, key, cipher_type="encrypt"):
    if not is_valid_key(key): return None

    if cipher_type == "encrypt":
        # מיפוי: ALPHABET[i] → key[i]
        table = {ALPHABET[i]: key[i] for i in range(N)}
    else:
        # הפוך: key[i] → ALPHABET[i]
        table = {key[i]: ALPHABET[i] for i in range(N)}

    return "".join(table.get(ch, ch) for ch in message)

def hack_simple_substitution(ciphertext: str):
    # ─ שלב 1: זיהוי תו הרווח ─
    space_char = detect_space_char(ciphertext)

    # ─ שלב 2: גשר A-Z + Word Patterns ─
    freq = Counter(ch for ch in ciphertext if ch != space_char)
    top26      = [ch for ch,_ in freq.most_common(26)]
    ext_to_az  = {ch: AZ[i] for i,ch in enumerate(top26)}
    cipher_az  = ''.join(ext_to_az.get(ch,' ') if ch != space_char
                          else ' ' for ch in ciphertext)

    az_mapping = build_cipher_text_letter_mapping(cipher_az, ' ')
    # תרגם חזרה לאלפבית המורחב
    mapping = translate_az_mapping_to_extended(az_mapping, ...)

    # ─ שלב 3: complete_mapping ─
    cracked = translate_using_mapping(mapping, ciphertext)
    if get_unmapped_keys(mapping):
        cracked, mapping = complete_mapping(
            ciphertext, cracked, mapping, space_char)

    return cracked, mapping
