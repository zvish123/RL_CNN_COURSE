"""screen_capture.py — Host side pipeline"""
import mss, io, time, asyncio
import numpy as np
from PIL import Image

class ScreenCapture:
    def __init__(self, quality=60, fps=30):
        self.sct     = mss.mss()
        self.quality = quality
        self.interval = 1.0 / fps
        self.prev_frame = None

    def capture_frame(self) -> bytes:
        # לכוד מסך
        monitor = self.sct.monitors[1]  # מסך ראשי
        raw = self.sct.grab(monitor)
        img = Image.frombytes(
            "RGB", raw.size, raw.bgra,
            "raw", "BGRX")
        return img

    def encode_frame(self, img) -> bytes | None:
        # Diff — שלח רק שינויים
        arr = np.array(img)
        if self.prev_frame is not None:
            diff = np.abs(
                arr.astype("int16") -
                self.prev_frame.astype("int16"))
            if diff.max() < 8:
                return None  # אין שינוי
        self.prev_frame = arr

        # דחוס ל-JPEG
        buf = io.BytesIO()
        img.save(buf, "JPEG",
                quality=self.quality,
                optimize=True)
        return buf.getvalue()

    async def stream(self,
                       writer: asyncio.StreamWriter):
        """שולח frames לclient דרך TCP"""
        while True:
            t0 = time.perf_counter()
            img  = self.capture_frame()
            data = self.encode_frame(img)
            if data:
                # 4-byte header: גודל ה-frame
                hdr = len(data).to_bytes(4,"big")
                writer.write(hdr + data)
                await writer.drain()
            elapsed = time.perf_counter() - t0
            await asyncio.sleep(
                max(0, self.interval - elapsed))

💡 מה הקוד עושה?

mss.grab() — לכידה מהירה (מהיר מ-PIL)
numpy diff — השוואה לframe קודם
JPEG encode — דחיסה עם quality שניתן לשנות
asyncio.sleep — throttle ל-FPS מטרה

"""input_handler.py — Host side input"""
import ctypes, ctypes.wintypes as wt
from dataclasses import dataclass
import json

MOUSEEVENTF_MOVE        = 0x0001
MOUSEEVENTF_LEFTDOWN    = 0x0002
MOUSEEVENTF_LEFTUP      = 0x0004
MOUSEEVENTF_ABSOLUTE    = 0x8000
INPUT_KEYBOARD          = 1
INPUT_MOUSE             = 0
KEYEVENTF_KEYUP         = 0x0002

k32 = ctypes.windll.kernel32
u32 = ctypes.windll.user32

def move_mouse(x: int, y: int,
               screen_w: int,
               screen_h: int):
    # המרה לקואורדינטות מוחלטות 0-65535
    ax = int(x / screen_w * 65535)
    ay = int(y / screen_h * 65535)
    u32.SetCursorPos(ax, ay)

def click_mouse(button='left',
                 down=True):
    flags = {
        ('left',  True):  MOUSEEVENTF_LEFTDOWN,
        ('left',  False): MOUSEEVENTF_LEFTUP,
    }
    f = flags.get((button, down), 0)
    u32.mouse_event(f, 0, 0, 0, 0)

def press_key(vk_code: int,
               key_up=False):
    flag = KEYEVENTF_KEYUP if key_up else 0
    u32.keybd_event(vk_code, 0, flag, 0)

def handle_input_event(msg: str):
    # מקבל JSON מה-Viewer
    data = json.loads(msg)
    etype = data['type']
    if etype == 'mouse_move':
        move_mouse(data['x'], data['y'],
                   data['sw'], data['sh'])
    elif etype == 'mouse_click':
        click_mouse(data['button'],
                    data['down'])
    elif etype == 'key':
        press_key(data['vk'],
                   data.get('up', False))

🗄️ DB Schema — Sessions & Events

השלם את ה-schema של ה-PostgreSQL ל-anyView:

import psycopg2, psycopg2.pool
from datetime import datetime

# Connection pool — מומלץ מאוד לPG
pool = psycopg2.pool.ThreadedConnectionPool(
    minconn=1, maxconn=10,
    host="localhost", dbname="anyview_db",
    user="anyview", password="SECRET"
)

def init_db():
    conn = pool.getconn()
    cur  = conn.cursor()
    cur.execute("""
        CREATE TABLE IF NOT EXISTS sessions (
            id         SERIAL PRIMARY KEY,
            host_ip    TEXT NOT NULL,
            viewer_ip  TEXT NOT NULL,
            started_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
            ended_at   TIMESTAMPTZ,
            status     TEXT DEFAULT ''
        )""")
    cur.execute("""
        CREATE TABLE IF NOT EXISTS events (
            id         SERIAL PRIMARY KEY,
            session_id INTEGER REFERENCES sessions(id),
            timestamp  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
            event_type TEXT,
              JSONB
        )""")
    cur.execute("""
        CREATE INDEX IF NOT EXISTS idx_events_session
            ON events()""")
    conn.commit()
    cur.close()
    pool.putconn(conn)

🔐 Security Architecture — TLS over TCP + Auth

"""server.py — asyncio TCP Server + TLS"""
import asyncio, ssl, json, jwt, bcrypt
from screen_capture import ScreenCapture
from input_handler  import handle_input_event

SECRET = "anyview-secret-CHANGE-ME"
VIDEO_PORT = 9000
INPUT_PORT = 9001

def make_tls_ctx() -> ssl.SSLContext:
    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_SERVER)
    ctx.load_cert_chain("cert.pem", "key.pem")
    ctx.minimum_version = ssl.TLSVersion.TLSv1_3
    return ctx

async def video_handler(
        reader: asyncio.StreamReader,
        writer: asyncio.StreamWriter):
    # 1. Auth handshake — קבל JWT
    token_line = await reader.readline()
    token = token_line.decode().strip()
    try:
        jwt.decode(token, SECRET,
                   algorithms=["HS256"])
    except Exception:
        writer.write(b"ERR:UNAUTHORIZED\n")
        await writer.drain()
        writer.close(); return

    writer.write(b"OK\n")
    await writer.drain()

    # 2. Stream frames
    cap = ScreenCapture(quality=60, fps=30)
    await cap.stream(writer)

async def input_handler(reader, writer):
    # קבל events JSON line-by-line
    async for line in reader:
        handle_input_event(line.decode())

async def main():
    ctx = make_tls_ctx()
    srv_v = await asyncio.start_server(
        video_handler, "0.0.0.0", VIDEO_PORT,
        ssl=ctx)
    srv_i = await asyncio.start_server(
        input_handler, "0.0.0.0", INPUT_PORT,
        ssl=ctx)
    print(f"Video :{VIDEO_PORT} | Input :{INPUT_PORT}")
    async with srv_v, srv_i:
        await asyncio.gather(
            srv_v.serve_forever(),
            srv_i.serve_forever())

asyncio.run(main())
