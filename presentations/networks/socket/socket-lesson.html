    socket.AF_INET,
    socket.SOCK_STREAM
)

server.listen(5)
print("שרת מאזין...")

conn, addr = server.accept()
print(f"לקוח: {addr}")

data = conn.recv(1024)
print(f"קיבלתי: {data.decode()}")
conn.send(b"שלום מהשרת!")
conn.close()

  app.py — pip install flask flask-sock
from flask import Flask
from flask_sock import Sock
import json

app = Flask(__name__)
sock = Sock(app)
clients = set()          # כל הלקוחות המחוברים

@sock.route('/ws')
def websocket(ws):
    clients.add(ws)
    try:
        while True:
            raw  = ws.receive()
            data = json.loads(raw)
            # שלח לכולם (broadcast)
            for client in clients.copy():
                try:
                    client.send(json.dumps(data))
                except:
                    clients.discard(client)
    finally:
        clients.discard(ws)

if __name__ == '__main__':
    app.run(port=5000)
  useWebSocket.js — Custom Hook
import { useState, useEffect, useRef } from 'react';

export function useWebSocket(url) {
  const [messages, setMessages] = useState([]);
  const [connected, setConnected] = useState(false);
  const wsRef = useRef(null);

  useEffect(() => {
    wsRef.current = new WebSocket(url);

    wsRef.current.onopen    = () => setConnected(true);
    wsRef.current.onclose   = () => setConnected(false);
    wsRef.current.onmessage = (e) => {
      const data = JSON.parse(e.data);
      setMessages(prev => [...prev, data]);
    };

    return () => wsRef.current?.close(); // cleanup!
  }, [url]);

  const send = (text) => {
    wsRef.current?.send(JSON.stringify({ text }));
  };

  return { messages, connected, send };
}

import asyncio, websockets, json, threading
from PyQt6.QtWidgets import *
from PyQt6.QtCore import pyqtSignal, QObject

class WSWorker(QObject):
    message_received = pyqtSignal(str)

    def __init__(self, url):
        super().__init__()
        self.url = url

    def start(self):
        threading.Thread(target=self._run, daemon=True).start()

    def _run(self):
        asyncio.run(self._connect())

    async def _connect(self):
        async with websockets.connect(self.url) as ws:
            self.ws = ws
            async for msg in ws:
                data = json.loads(msg)
                self.message_received.emit(data['text'])

# --- UI ---
class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.worker = WSWorker('ws://localhost:5000/ws')
        self.worker.message_received.connect(self.on_message)
        self.worker.start()

    def on_message(self, text):
        print(f"קיבלתי: {text}")

from flask import Flask
from flask_socketio import SocketIO, emit, join_room

app = Flask(__name__)
socketio = SocketIO(app, cors_allowed_origins="*")

@socketio.on('connect')
def on_connect():
    print("התחבר לקוח חדש")

@socketio.on('chat:message')   # ← שם ה-event
def on_message(data):
    # שלח לכולם
    emit('chat:message', data, broadcast=True)

@socketio.on('join')
def on_join(data):
    join_room(data['room'])             # ← rooms!
    emit('chat:message',
         {'text': f"הצטרף לחדר {data['room']}"},
         room=data['room'])

socketio.run(app, port=5000)

  pyqt_sio.py — pip install python-socketio[client] PyQt6
import socketio, threading
from PyQt6.QtWidgets import *
from PyQt6.QtCore import pyqtSignal, QObject

class SIOWorker(QObject):
    message_received = pyqtSignal(dict)

    def __init__(self):
        super().__init__()
        self.sio = socketio.Client()

        @self.sio.on('chat:message')
        def on_msg(data):
            self.message_received.emit(data)

    def start(self):
        threading.Thread(
            target=self.sio.connect,
            args=('http://localhost:5000',),
            daemon=True
        ).start()

    def send(self, text):
        self.sio.emit('chat:message', {'text': text})
