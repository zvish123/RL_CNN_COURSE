        def 
        create_linear_data(n_samples=1000, noise_std=0.1):
                    x = np.linspace(-5, 5, n_samples)
                    y = 2 * x + 1 + np.random.normal(0, noise_std, n_samples)
        return x.reshape(-1, 1), y.reshape(-1, 1)
        # יצירת נתונים לפונקציה ריבועית
        def 
        create_quadratic_data(n_samples=1000, noise_std=0.5):
                    x = np.linspace(-5, 5, n_samples)
                    y = 2 * x**2 + 3 * x + 5 + np.random.normal(0, noise_std, n_samples)
        return x.reshape(-1, 1), y.reshape(-1, 1)

🎛️ כוונון היפרפרמטרים

טבלת היפרפרמטרים לניסוי:

היפרפרמטר ערכים מומלצים השפעה טיפים
Learning Rate 0.001, 0.01, 0.1 מהירות הלמידה התחילו מ-0.001
Batch Size 16, 32, 64 יציבות הלמידה קטן יותר = יציב יותר
Epochs 100-500 זמן אימון עצרו כש-val_loss עולה
Hidden Units 8, 16, 32, 64 כושר ביטוי יותר = מורכב יותר
Dropout Rate 0.1, 0.2, 0.3 מניעת Overfitting רק אם יש Overfitting

🚨 סימני Overfitting

  • Training Loss ממשיך לרדת
  • Validation Loss עולה או קבוע
  • פער גדול בין Train ל-Validation
  • R² גבוה על Train, נמוך על Test

🔧 פתרונות

  • Dropout: 0.1-0.3
  • Early Stopping: עצירה כשיש Overfitting
  • רשת קטנה יותר: פחות פרמטרים
  • יותר נתונים: אם אפשר

קוד לניהול ניסויים:

        import pandas 
        as pd


        def 
        run_experiment(config, X_train, X_test, y_train, y_test):
        """
    מריץ ניסוי בודד עם קונפיגורציה נתונה
    """
        # בניית המודל
    model = keras.Sequential()
    model.add(layers.Dense(config[
        'hidden_units'], 
                          activation=
        'relu', 
                          input_shape=(1,)))

    
        for i 
        in range(config[
        'num_layers'] - 1):
        if config[
        'dropout_rate'] > 0:
            model.add(layers.Dropout(config[
        'dropout_rate']))
        model.add(layers.Dense(config[
        'hidden_units'] // (i+1), 
                              activation=
        'relu'))

    model.add(layers.Dense(1))

    
        # קומפילציה
    model.compile(
        optimizer=keras.optimizers.Adam(learning_rate=config[
        'learning_rate']),
        loss=
        'mse',
        metrics=[
        'mae']
    )

    
        # אימון
    history = model.fit(
        X_train, y_train,
        epochs=config[
        'epochs'],
        batch_size=config[
        'batch_size'],
        validation_split=0.2,
        verbose=0
    )

    
        # הערכה
    y_pred = model.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    r2 = r2_score(y_test, y_pred)

    
        return {
        'mse': mse,
        'r2': r2,
        'final_train_loss': history.history[
        'loss'][-1],
        'final_val_loss': history.history[
        'val_loss'][-1],
        'model': model
    }

ניתוח התוצאות:

        def 
        analyze_results(df_results):
        """
    מנתח את התוצאות ומדפיס תובנות
    """
    best_model = df_results.iloc[0]

    print(
        "🏆 המודל הטוב ביותר:")
    print(f
        "   שם: {best_model['name']}")
    print(f
        "   R²: {best_model['r2']:.4f}")
    print(f
        "   MSE: {best_model['mse']:.4f}")

    
        # בדיקת Overfitting
        for _, row 
        in df_results.iterrows():
        train_val_ratio = row[
        'final_train_loss'] / row[
        'final_val_loss']
        if train_val_ratio < 0.5:
            print(f
        "⚠️  {row['name']} מראה סימני Overfitting")
        elif train_val_ratio > 0.9:
            print(f
        "✅ {row['name']} מאוזן טוב")

    
        return best_model

best_model = analyze_results(df_results)
