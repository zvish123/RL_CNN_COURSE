שתי שכבות אבטחה

def create_token(username):
    user = USERS_DB[username]
    payload = {
        "user_id":     username,
        "role":        user["role"],
        "permissions": user["permissions"],
        "scope":       user["scope"],
        "resources":   user["resources"],
        "exp": datetime.utcnow() + timedelta(hours=1),
    }
    return jwt.encode(payload, SECRET_KEY,
                       algorithm="HS256")

// טיפול בלקוח (per-thread)

def handle_client(conn, addr):
  with conn:
    # שלב 1: שלח ready
    send_json(conn, {"status":"ready"})

    # שלב 2: קבל credentials
    raw = recv_line(conn)
    username, password = raw.split(":", 1)
    user = USERS_DB.get(username)
    if not user or user["password"] != password:
        send_json(conn, {"status":"error"})
        return

    # שלב 3: הנפק JWT
    token = create_token(username)
    send_json(conn, {"status":"ok",
                     "token": token})

    # שלב 4: לולאת בקשות
    while True:
        req = json.loads(recv_line(conn))
        payload = decode_token(req["token"])
        # בדוק הרשאה → שלח תוצאה
        result = dispatch_check(payload, req)
        send_json(conn, result)
04 — קוד לקוח

// חיבור TLS

def make_tls_connection():
    ctx = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
    # אימות אישור השרת
    ctx.load_verify_locations("server.crt")
    # self-signed → ללא hostname check
    ctx.check_hostname = False

    raw = socket.socket(AF_INET, SOCK_STREAM)
    # עוטפים ב-TLS
    tls = ctx.wrap_socket(raw,
              server_hostname=HOST)
    tls.connect((HOST, PORT))
    return tls

// login + קבלת JWT

with make_tls_connection() as conn:
    # הדפס cipher שנבחר
    cipher = conn.cipher()
    # קבל ready

        f"{username}:{password}\n".encode()

05 — בדיקות הרשאה

4 פונקציות בדיקה

# 1. RBAC — רמת תפקיד
def check_rbac(payload, required_role):
    levels = {"admin":3,"editor":2,"viewer":1}
    return (levels.get(payload["role"],0)
            >= levels.get(required_role,99))

# 2. Permission-Based — פעולה ספציפית
def check_permission(payload, action):
    return action in payload.get(
               "permissions", [])
# 3. Scope — OAuth2 סגנון
#    read:all מכסה read:reports
def check_scope(payload, required_scope):
    scopes = payload.get("scope","").split()
    if required_scope in scopes:
        return True
    action = required_scope.split(":")[0]
    return f"{action}:all" in scopes

# 4. Resource-Based — הרשאה למשאב
def check_resource(payload, resource, action):
    levels = {"admin":3,"write":2,"read":1}
    user_level = levels.get(
        payload.get("resources",{})
               .get(resource,""), 0)
    return user_level >= levels.get(action,99)

06 — תעבורת רשת

מה עובר בתעבורה?

עיקרון: ה-TLS עוטף את כל התקשורת — כולל credentials, טוקנים ותוצאות. מאזין על הקו רואה רק bytes מוצפנים.
שלב מה נשלח (plaintext לוגי) מה נראה על הרשת כיוון
TLS Handshake ClientHello, ServerHello, Certificate, Finished TLS records — metadata גלוי, תוכן מוצפן ⇄
Ready {"status":"ready","msg":"שלח username:password"} 🔒 AES-256-GCM ciphertext ← שרת
Login alice:alice_pass\n 🔒 AES-256-GCM ciphertext → לקוח
JWT Token {"status":"ok","token":"eyJhbGci..."} 🔒 AES-256-GCM ciphertext ← שרת
בקשת הרשאה {"token":"eyJ...","check":"rbac","value":"admin"} 🔒 AES-256-GCM ciphertext → לקוח
תשובת הרשאה {"status":"OK","check":"RBAC role>=admin","user":"alice"} 🔒 AES-256-GCM ciphertext ← שרת
