שרת בסיסי

import socket

HOST = "127.0.0.1"
PORT = 5000

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
server.bind((HOST, PORT))
server.listen(1)

client_socket, addr = server.accept()
data = client_socket.recv(1024)
print(data.decode())
client_socket.sendall("hello".encode())

client_socket.close()
server.close()

לקוח בסיסי

import socket

HOST = "127.0.0.1"
PORT = 5000

client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client.connect((HOST, PORT))
client.sendall("hi".encode())
reply = client.recv(1024)
print(reply.decode())
client.close()

הפקודות הכי חשובות

פקודה מה היא עושה הערה קצרה
socket() יוצרת socket חדש נקודת ההתחלה
bind((host, port)) מחברת שרת לכתובת ולפורט לשרת בלבד
listen(n) מעבירה שרת למצב האזנה לפני accept()
accept() מחכה לחיבור ומחזירה socket חדש מחזירה גם כתובת לקוח
connect((host, port)) מחברת לקוח לשרת ללקוח בלבד
recv(1024) מקבלת עד 1024 בתים מחזירה bytes
send() שולחת נתונים לא תמיד שולחת הכל
sendall() שולחת את כל הנתונים בדרך כלל עדיפה
close() סוגרת socket חשוב מאוד

encode / decode

Socket עובד עם bytes, לא עם str.

message = "hello"
client.sendall(message.encode())

data = client.recv(1024)
