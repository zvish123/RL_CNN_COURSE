                num_games=int(experiment['num_games']),
                gamma=float(experiment['gamma']),
                exploration_rate=float(
                    experiment['exploration_rate']
                ),
                loss_penalty=float(
                    experiment['loss_penalty']
                )
            )
            experiments.append(config)
    
    return experiments
