def run_single_experiment(self, config, save_results=True):
    # 1. התחלת מדידת זמן
    start_time = time.time()
    
    # 2. יצירת טורניר עם הקונפיגורציה
    tournament = XOTournament(config)
    
    # 3. הרצת הטורניר
    tournament.run_tournament()
    
    # 4. חישוב זמן ריצה
    runtime = time.time() - start_time
    
    # 5. איסוף סטטיסטיקות
    stats = tournament.get_statistics()
    stats['runtime_seconds'] = runtime
    stats['experiment_name'] = config.name
    
    # 6. שמירת תוצאות
    if save_results:
        tournament.save_results(filename)
    
    return stats

def run_multiple_experiments(self, configs, repetitions=1):
    """
    :param configs: רשימת תצורות ניסויים
    :param repetitions: מספר חזרות לכל ניסוי
    """
    all_results = []
    total_experiments = len(configs) * repetitions
    
    for config in configs:
        for rep in range(repetitions):
            # שינוי שם אם יש חזרות
            if repetitions > 1:
                config.name = f"{original_name}_rep{rep + 1}"
            
            # הרצת ניסוי בודד
            result = self.run_single_experiment(config)
            all_results.append(result)
    
    # שמירת סיכום כולל
    self.save_summary()
    return all_results

🔄 למה חזרות (repetitions)?

כיוון שיש אקראיות במשחק, תוצאות עשויות להשתנות. חזרות מרובות נותנות תוצאות אמינות יותר ומאפשרות לחשב סטיית תקן.

⚙️ ExperimentConfig — שדות חדשים

פרמטרים שנוספו לשליטה מתקדמת יותר

@dataclass
class ExperimentConfig:
    name: str
    description: str
    num_games: int = 1000
    exploration_rate: float = 0.1
    gamma: float = 0.9
    win_reward: float = 1.0
    loss_penalty: float = 0.0
    tie_reward: float = 0.5
    blocking_reward: float = 0.95  # 🆕
    rival_plan: str = "random"      # 🆕
    mixed_plan_ratio = [0.2]          # 🆕

🎯 rival_plan — ערכים אפשריים

  • "random" — O משחק אקראי לחלוטין
  • "heuristic" — O משחק חכם (חוסם, מנצח)
  • "mixed" — מתחיל אקראי, עובר להיוריסטיקה

🛡️ blocking_reward = 0.95

🔀 mixed_plan_ratio = [0.2]

ה-AI מתאמן תחילה נגד יריב אקראי ב-20% הראשונים של המשחקים, לאחר מכן עובר ליריב ההיוריסטי החכם יותר

def load_experiments(self, experiments_csv):
    experiments = []
    with open(experiments_csv, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            config = ExperimentConfig(
                name=row['experiment_name'],
                description=row['description'],
                num_games=int(row['num_games']),
                gamma=float(row['gamma']),
                exploration_rate=float(row['exploration_rate']),
                win_reward=float(row['win_reward']),
                loss_penalty=float(row['loss_penalty']),
                tie_reward=float(row['tie_reward']),
                blocking_reward=float(row['blocking_reward'])
            )
            experiments.append(config)
    return experiments
