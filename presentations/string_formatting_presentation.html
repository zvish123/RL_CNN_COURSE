למה זה חשוב? 💡

  • קריאות: קל יותר להבין מה התוכנית עושה
  • מקצועיות: נראה כמו תוכנה אמיתית
  • דיבוג: קל יותר למצוא בעיות
  • בגרות: ציון טוב יותר במדעי המחשב!

def check_grade(grade):
    if grade < 0 or grade > 100:
        print(f"❌ שגיאה: הציון {grade} לא תקין (צריך להיות בין 0-100)")
        return False
    elif grade >= 90:
        print(f"🏆 מעולה! הציון שלך {grade} - המשך כך!")
        return True
    elif grade >= 80:
        print(f"👍 טוב מאוד! הציון שלך {grade}")
        return True
    elif grade >= 60:
        print(f"📝 בסדר. הציון שלך {grade} - אפשר לשפר")
        return True
    else:
        print(f"😞 הציון שלך {grade} - בואו נעבוד יותר קשה")
        return True

# דוגמאות

import time

def show_progress(current, total, task=""):
    percentage = (current / total) * 100
    bar_length = 20
    filled = int(bar_length * current // total)
    bar = '█' * filled + '░' * (bar_length - filled)
    
    print(f"\r{task} [{bar}] {percentage:.1f}% ({current}/{total})", end="")
    if current == total:
        print(" ✅ הושלם!")

# דוגמה
print("מוריד קובץ...")
for i in range(1, 11):
    show_progress(i, 10, "הורדה")
    time.sleep(0.1)  # סימולציה של זמן המתנה

def calculate_bagrut(internal_grade, external_grade, coefficient=1):
    final_grade = (internal_grade * 0.25) + (external_grade * 0.75)
    points = final_grade * coefficient
    
    print("🎓 חישוב ציון בגרות")
    print("-" * 25)
    print(f"ציון פנימי (25%): {internal_grade}")
    print(f"ציון חיצוני (75%): {external_grade}")
    print(f"מקדם: {coefficient}")
    print(f"ציון סופי: {final_grade:.1f}")
    print(f"נקודות זכות: {points:.1f}")
    
    if points >= 450:  # 90*5
        print("🏆 מצוינות!")
    elif points >= 350:  # 70*5
        print("👍 טוב מאוד!")
    else:
        print("💪 יש מקום לשיפור")
    
    return final_grade, points

def take_attendance(students):
    attendance = {}
    print("📋 רישום נוכחות")
    print("=" * 20)
    
    for student in students:
        while True:
            status = input(f"{student} (נ/ח/א): ").lower()
            if status in ['נ', 'ח', 'א']:
                attendance[student] = {
                    'נ': 'נוכח', 'ח': 'חסר', 'א': 'איחור'
                }[status]
                break
            print("❌ הכניסו נ/ח/א בלבד")
    
    # דוח
    print(f"\n📊 דוח נוכחות - {len(students)} תלמידים")
    print("-" * 30)
    for student, status in attendance.items():
        emoji = {'נוכח': '✅', 'חסר': '❌', 'איחור': '⏰'}[status]
        print(f"{emoji} {student:<8} {status}")
    
    # סטטיסטיקות
    present = sum(1 for s in attendance.values() if s == 'נוכח')
    absent = sum(1 for s in attendance.values() if s == 'חסר')
    late = sum(1 for s in attendance.values() if s == 'איחור')
    
    print(f"\n📈 סיכום:")
    print(f"נוכחים: {present}, חסרים: {absent}, איחורים: {late}")
    print(f"אחוז נוכחות: {(present/len(students))*100:.1f}%")

def format_grade(grade):
    color_code = "🟢" if grade >= 90 else "🟡" if grade >= 70 else "🔴"
    status = "מעולה" if grade >= 90 else "טוב" if grade >= 70 else "משפר"
    return f"{color_code} {grade} ({status})"

grades = [95, 67, 78, 91]
for i, grade in enumerate(grades, 1):
    print(f"תלמיד {i}: {format_grade(grade)}")

3. פרמוט עם Locale (עברית):

import locale
from datetime import datetime

# הגדרת עברית (אם זמין במערכת)
try:
    locale.setlocale(locale.LC_TIME, 'he_IL.UTF-8')
    now = datetime.now()
    print(f"היום: {now.strftime('%A, %d %B %Y')}")
except:
    # חלופה ללא locale
    hebrew_months = {
        1: "ינואר", 2: "פברואר", 3: "מרץ", 4: "אפריל",
        5: "מאי", 6: "יוני", 7: "יולי", 8: "אוגוסט",
        9: "ספטמבר", 10: "אוקטובר", 11: "נובמבר", 12: "דצמבר"
    }
    now = datetime.now()
    month_name = hebrew_months[now.month]
    print(f"היום: {now.day} ב{month_name} {now.year}")

def get_valid_grade():
    while True:
        try:
            grade = float(input("הכניסו ציון (0-100): "))
            if 0 <= grade <= 100:
                return grade
            else:
                print("❌ ציון צריך להיות בין 0-100")
        except ValueError:
            print("❌ אנא הכניסו מספר תקין")
