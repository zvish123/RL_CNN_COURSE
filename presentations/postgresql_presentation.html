🏗️ יצירת טבלאות

בואו ניצור טבלת תלמידים לפרויקט:

CREATE TABLE students (
    student_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE,
    birth_date DATE,
    grade_level INTEGER CHECK (grade_level BETWEEN 7 AND 12)
);
🔍 הסבר על הקוד:
  • SERIAL - מספר שעולה אוטומטית (1, 2, 3...)
  • PRIMARY KEY - מזהה ייחודי לכל תלמיד
  • VARCHAR(50) - טקסט עד 50 תווים
  • NOT NULL - שדה חובה
  • UNIQUE - אין שני אימיילים זהים
  • CHECK - בדיקת תקינות (כיתות 7-12)

📝 סוגי נתונים נפוצים

סוג מתי משתמשים דוגמה
INTEGER מספרים שלמים גיל: 16, כמות: 5
VARCHAR(n) טקסט באורך משתנה שם: "יוסי", עיר: "תל אביב"
TEXT טקסט ארוך תיאור, הערות
DATE תאריך 2024-01-15
BOOLEAN אמת/שקר is_active: TRUE
DECIMAL(p,s) מספרים עשרוניים ציון: 95.5, מחיר: 19.90
🎯 שאלת חשיבה: איזה סוג נתונים תבחרו?
לשמירת מספר טלפון: INTEGER
לשמירת מספר טלפון: BOOLEAN
לשמירת מספר טלפון: VARCHAR(20)

💡 רמז: מספר טלפון יכול להתחיל ב-0 ויכול לכלול מקפים

➕ הוספת נתונים

תלמיד אחד:

INSERT INTO students (first_name, last_name, email, grade_level)
VALUES ('יוסי', 'כהן', 'yossi@school.com', 10);

כמה תלמידים ביחד:

INSERT INTO students (first_name, last_name, email, grade_level)
VALUES 
    ('מירי', 'לוי', 'miri@school.com', 11),
    ('דני', 'אברהם', 'dani@school.com', 10),
    ('שרה', 'ישראלי', 'sarah@school.com', 12);
✅ טיפ חשוב:
  • אין צורך למלא את student_id - הוא נוצר אוטומטית!
  • שימו לב לסימני המרכאה הבודדות (') סביב טקסט
  • מספרים בלי מרכאות

🔍 שליפת נתונים - SELECT

שליפת כל התלמידים:

SELECT * FROM students;

שליפת עמודות ספציפיות:

SELECT first_name, last_name, email 
FROM students;

שליפה עם תנאי:

-- תלמידי כיתה י'
SELECT * FROM students 
WHERE grade_level = 10;

-- תלמידים שהשם שלהם מתחיל ב"י"
SELECT * FROM students 
WHERE first_name LIKE 'י%';

-- מיון לפי שם משפחה
SELECT * FROM students 
ORDER BY last_name;
💻 תרגיל:

איך תשלפו את כל התלמידים מכיתות יא' ויב'?

SELECT * FROM students
WHERE grade_level IN (11, 12);

-- או:
SELECT * FROM students
WHERE grade_level >= 11;

✏️ עדכון נתונים

-- עדכון אימייל של תלמיד
UPDATE students
SET email = 'new.email@school.com'
WHERE student_id = 1;

-- העלאת כיתה לכל התלמידים (בסוף שנה)
UPDATE students
SET grade_level = grade_level + 1
WHERE grade_level < 12;
⚠️ אזהרה חשובה!

אם תשכחו את WHERE, תעדכנו את כל השורות בטבלה!

-- זה ימחק את כל האימיילים! 😱
UPDATE students
SET email = NULL;

-- תמיד השתמשו ב-WHERE!
UPDATE students
SET email = NULL
WHERE student_id = 5;

🗑️ מחיקת נתונים

-- מחיקת תלמיד ספציפי
DELETE FROM students
WHERE student_id = 5;

-- מחיקת כל תלמידי כיתה יב' (אחרי סיום)
DELETE FROM students
WHERE grade_level = 12;
🚨 סכנה! קראו היטב! 
-- זה ימחק את כל הטבלה! 💀
DELETE FROM students;

-- תמיד תשאלו את עצמכם:
-- "האם אני רוצה למחוק את כל הנתונים?"
-- אם לא - הוסיפו WHERE!

✅ נכון

תמיד בדקו קודם:

SELECT * FROM students

WHERE student_id = 5;


-- אם זה מה שרציתם:

DELETE FROM students

WHERE student_id = 5;

❌ מסוכן

למחוק בלי לבדוק:

DELETE FROM students;

🔗 קשרים בין טבלאות

הכוח של בסיסי נתונים: חיבור טבלאות!

דוגמה: מערכת קורסים

-- טבלת קורסים
CREATE TABLE courses (
    course_id SERIAL PRIMARY KEY,
    course_name VARCHAR(100) NOT NULL,
    teacher_name VARCHAR(100)
);

-- טבלת הרשמות (קושרת תלמידים לקורסים)
CREATE TABLE enrollments (
    enrollment_id SERIAL PRIMARY KEY,
    student_id INTEGER REFERENCES students(student_id),
    course_id INTEGER REFERENCES courses(course_id),
    grade DECIMAL(4,2)
);
🎯 למה זה חשוב?
  • תלמיד אחד יכול להירשם לכמה קורסים
  • קורס אחד יכול להכיל כמה תלמידים
  • לא משכפלים מידע (שם התלמיד מופיע פעם אחת בלבד)
  • REFERENCES מבטיח שלא נוכל להרשם לקורס שלא קיים!

🤝 JOIN - חיבור טבלאות

רוצים לראות אילו קורסים כל תלמיד לומד?

SELECT 
    s.first_name,
    s.last_name,
    c.course_name,
    e.grade
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
JOIN courses c ON e.course_id = c.course_id
ORDER BY s.last_name;

תוצאה:

שם פרטי שם משפחה קורס ציון
יוסי כהן מתמטיקה 95.5
יוסי כהן אנגלית 88.0
מירי לוי היסטוריה 92.0
💡 הסבר: JOIN מאפשר לנו לחבר מידע משלוש טבלאות שונות לשאילתה אחת!

📊 פונקציות אגרגציה

רוצים סטטיסטיקות? יש לנו!

-- כמה תלמידים יש בכל כיתה?
SELECT grade_level, COUNT(*) AS student_count
FROM students
GROUP BY grade_level;

-- מה הציון הממוצע של כל תלמיד?
SELECT 
    s.first_name,
    s.last_name,
    AVG(e.grade) AS average_grade,
    COUNT(e.course_id) AS num_courses
FROM students s
JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.student_id, s.first_name, s.last_name;

🐍 חיבור ל-Python

זה מה שתשתמשו בפרויקט שלכם!

התקנת הספרייה:

pip install psycopg2

קוד לדוגמה:

import psycopg2

# התחברות
conn = psycopg2.connect(
    dbname="my_school",
    user="postgres",
    password="your_password",
    host="localhost",
    port="5432"
)

cursor = conn.cursor()

# הוספת תלמיד
cursor.execute("""
    INSERT INTO students (first_name, last_name, grade_level)
    VALUES (%s, %s, %s)
""", ("יוסי", "כהן", 10))

# שמירה
conn.commit()

# שליפת נתונים
cursor.execute("SELECT * FROM students")
students = cursor.fetchall()

for student in students:
    print(student)

cursor.close()
conn.close()
⚠️ חשוב: השתמשו ב-%s ולא ב-f-strings למניעת SQL Injection!

  1. Query Tool
    כתבו שאילתות והריצו אותן בקלות
  2. View/Edit Data
    ערכו נתונים ישירות בטבלה
  3. ER Diagram
    ראו תרשים של הקשרים בין הטבלאות
