"""db.py — Database manager for WinAPI Monitor"""
import sqlite3
from datetime import datetime
from pathlib import Path

DB_PATH = Path(__file__).parent / "monitor.db"

def get_conn():
    """מחזיר חיבור ל-DB עם Row Factory"""
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    """יצירת כל הטבלאות אם לא קיימות"""
    conn = get_conn()
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS windows (
            id           INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp    TEXT NOT NULL,
            window_title TEXT,
            process_name TEXT,
            pid          INTEGER
        );
        CREATE TABLE IF NOT EXISTS processes (
            id        INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp TEXT NOT NULL,
            pid       INTEGER,
            name      TEXT,
            path      TEXT
        );
        CREATE TABLE IF NOT EXISTS clipboard (
            id           INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp    TEXT NOT NULL,
            content      TEXT,
            content_type TEXT
        );
        CREATE TABLE IF NOT EXISTS system_snapshots (
            id            INTEGER PRIMARY KEY AUTOINCREMENT,
            timestamp     TEXT NOT NULL,
            computer_name TEXT,
            username      TEXT,
            is_admin      INTEGER,
            process_count INTEGER
        );
        CREATE INDEX IF NOT EXISTS idx_win_ts
            ON windows(timestamp);
        CREATE INDEX IF NOT EXISTS idx_proc_name
            ON processes(name);
    """)
    conn.commit()
    conn.close()
    print(f"[DB] Initialized: {DB_PATH}")

def insert_window(title: str, proc_name: str, pid: int):
    ts = datetime.now().isoformat(timespec='seconds')
    conn = get_conn()
    conn.execute(
        "INSERT INTO windows(timestamp,window_title,process_name,pid)"
        "VALUES(?,?,?,?)",
        (ts, title, proc_name, pid)
    )
    conn.commit(); conn.close()

def insert_processes(procs: list):
    ts = datetime.now().isoformat(timespec='seconds')
    conn = get_conn()
    conn.executemany(
        "INSERT INTO processes(timestamp,pid,name,path) VALUES(?,?,?,?)",
        [(ts, p['pid'], p['name'], p['path']) for p in procs]
    )
    conn.commit(); conn.close()

def insert_clipboard(content: str, ctype: str):
    ts = datetime.now().isoformat(timespec='seconds')
    conn = get_conn()
    conn.execute(
        "INSERT INTO clipboard(timestamp,content,content_type) VALUES(?,?,?)",
        (ts, content[:500], ctype)
    )
    conn.commit(); conn.close()

def insert_snapshot(computer: str, user: str,
                      admin: bool, proc_count: int):
    ts = datetime.now().isoformat(timespec='seconds')
    conn = get_conn()
    conn.execute(
        "INSERT INTO system_snapshots"
        "(timestamp,computer_name,username,is_admin,process_count)"
        "VALUES(?,?,?,?,?)",
        (ts, computer, user, 1 if admin else 0, proc_count)
    )
    conn.commit(); conn.close()

if __name__ == "__main__":
    init_db()
    print("DB schema created ✓")

"""collectors/windows.py — Active Window Monitor"""
import ctypes, ctypes.wintypes as wt, time
import db

user32 = ctypes.windll.user32
kernel32 = ctypes.windll.kernel32
psapi   = ctypes.windll.psapi

def get_active_window() -> dict:
    """מחזיר dict עם פרטי החלון הפעיל"""
    hwnd = user32.GetForegroundWindow()
    if not hwnd:
        return {}

    # קבל כותרת חלון
    title_buf = ctypes.create_unicode_buffer(512)
    user32.GetWindowTextW(hwnd, title_buf, 512)
    title = title_buf.value

    # קבל PID של התהליך שמחזיק את החלון
    pid = wt.DWORD()
    user32.GetWindowThreadProcessId(hwnd, ctypes.byref(pid))

    # פתח handle לprocess וקבל את שמו
    h_proc = kernel32.OpenProcess(
        0x0400 | 0x0010,  # PROCESS_QUERY_INFO | PROCESS_VM_READ
        False,
        pid.value
    )
    proc_name = "unknown"
    if h_proc:
        name_buf = ctypes.create_unicode_buffer(260)
        psapi.GetModuleBaseNameW(h_proc, None, name_buf, 260)
        proc_name = name_buf.value or "unknown"
        kernel32.CloseHandle(h_proc)   # תמיד לסגור!

    return {
        "hwnd":    hwnd,
        "title":   title,
        "pid":     pid.value,
        "process": proc_name,
    }

def run_window_tracker(interval: float = 2.0,
                         stop_event=None):
    """
    מריץ לולאת ניטור — שומר לDB כל פעם שהחלון משתנה.
    stop_event: threading.Event לעצירה מסודרת.
    """
    last_title = ""
    print("[WindowTracker] Started")

    while not (stop_event and stop_event.is_set()):
        info = get_active_window()
        if info and info["title"] != last_title:
            last_title = info["title"]
            db.insert_window(
                info["title"],
                info["process"],
                info["pid"]
            )
            print(f"  [WIN] {info['process']} → {info['title'][:60]}")
        time.sleep(interval)

    print("[WindowTracker] Stopped")

WinAPI: EnumProcesses, OpenProcess, GetModuleBaseName, QueryFullProcessImageName

"""collectors/processes.py — Process List Scanner"""
import ctypes, ctypes.wintypes as wt, time
import db

kernel32 = ctypes.windll.kernel32
psapi    = ctypes.windll.psapi

def get_all_processes() -> list:
    """מחזיר list של כל ה-processes עם PID, שם ומסלול"""
    pids = (wt.DWORD * 1024)()
    needed = wt.DWORD()
    psapi.EnumProcesses(pids, ctypes.sizeof(pids),
                         ctypes.byref(needed))
    count = needed.value // ctypes.sizeof(wt.DWORD)

    results = []
    for i in range(count):
        pid = pids[i]
        if pid == 0: continue

        h = kernel32.OpenProcess(
            0x0400 | 0x0010 | 0x1000,  # QUERY + VM_READ + LIMITED
            False, pid
        )
        name, path = "unknown", ""
        if h:
            # שם קצר
            nb = ctypes.create_unicode_buffer(260)
            psapi.GetModuleBaseNameW(h, None, nb, 260)
            name = nb.value or "unknown"

            # מסלול מלא
            pb   = ctypes.create_unicode_buffer(1024)
            size = wt.DWORD(1024)
            kernel32.QueryFullProcessImageNameW(h, 0, pb,
                                                 ctypes.byref(size))
            path = pb.value
            kernel32.CloseHandle(h)

        results.append({"pid": pid, "name": name, "path": path})

    return results

def run_process_scanner(interval: float = 30.0,
                          stop_event=None):
    """סורק processes כל interval שניות ושומר לDB"""
    print("[ProcessScanner] Started")
    while not (stop_event and stop_event.is_set()):
        procs = get_all_processes()
        db.insert_processes(procs)
        print(f"  [PROC] Scanned {len(procs)} processes")
        time.sleep(interval)
    print("[ProcessScanner] Stopped")

"""collectors/clipboard.py — Clipboard Monitor"""
import ctypes, ctypes.wintypes as wt, time, re
import db

user32   = ctypes.windll.user32
kernel32 = ctypes.windll.kernel32

CF_UNICODETEXT = 13

def read_clipboard() -> str:
    """קורא את תוכן ה-Clipboard. מחזיר מחרוזת ריקה אם נכשל."""
    if not user32.OpenClipboard(0):
        return ""
    try:
        h = user32.GetClipboardData(CF_UNICODETEXT)
        if not h:
            return ""
        ptr = kernel32.GlobalLock(h)
        if not ptr:
            return ""
        text = ctypes.wstring_at(ptr)
        kernel32.GlobalUnlock(h)
        return text
    finally:
        user32.CloseClipboard()

def classify_content(text: str) -> str:
    """מסווג תוכן clipboard לפי pattern"""
    if re.search(r'https?://', text):       return "url"
    if re.search(r'[\w.]+@[\w.]+\.\w+', text): return "email"
    if re.search(r'password|passwd|pwd',
                  text, re.IGNORECASE):       return "credential"
    if re.search(r'\b\d{4}[\s-]?\d{4}[\s-]?\d{4}[\s-]?\d{4}\b',
                  text):                     return "credit_card"
    return "text"

def run_clipboard_monitor(interval: float = 3.0,
                            stop_event=None):
    """מנטר clipboard ושומר לDB רק כשיש שינוי"""
    last = ""
    print("[ClipboardMon] Started")
    while not (stop_event and stop_event.is_set()):
        text = read_clipboard()
        if text and text != last:
            last = text
            ctype = classify_content(text)
            db.insert_clipboard(text, ctype)
            print(f"  [CLIP] [{ctype.upper()}] {text[:50]}..."
                  if len(text) > 50 else
                  f"  [CLIP] [{ctype.upper()}] {text}")
        time.sleep(interval)
    print("[ClipboardMon] Stopped")

"""collectors/sysinfo.py — System Snapshot Collector"""
import ctypes, ctypes.wintypes as wt, time
import db

kernel32 = ctypes.windll.kernel32
shell32  = ctypes.windll.shell32
psapi    = ctypes.windll.psapi

def get_snapshot() -> dict:
    """מצלם מצב המערכת ברגע זה"""
    buf  = ctypes.create_unicode_buffer(256)
    size = wt.DWORD(256)

    # שם מחשב
    kernel32.GetComputerNameW(buf, ctypes.byref(size))
    computer = buf.value

    # שם משתמש
    size.value = 256
    kernel32.GetUserNameW(buf, ctypes.byref(size))
    user = buf.value

    # הרשאות Admin
    is_admin = bool(shell32.IsUserAnAdmin())

    # כמות processes
    pids   = (wt.DWORD * 1024)()
    needed = wt.DWORD()
    psapi.EnumProcesses(pids, ctypes.sizeof(pids),
                         ctypes.byref(needed))
    proc_count = needed.value // ctypes.sizeof(wt.DWORD)

    return {
        "computer":   computer,
        "user":       user,
        "is_admin":   is_admin,
        "proc_count": proc_count,
    }

def run_sysinfo(interval: float = 60.0, stop_event=None):
    """שומר snapshot של המערכת כל interval שניות"""
    print("[SysInfo] Started")
    while not (stop_event and stop_event.is_set()):
        snap = get_snapshot()
        db.insert_snapshot(snap["computer"], snap["user"],
                            snap["is_admin"], snap["proc_count"])
        print(f"  [SYS] {snap['user']}@{snap['computer']} | "
              f"Admin:{snap['is_admin']} | Procs:{snap['proc_count']}")
        time.sleep(interval)
    print("[SysInfo] Stopped")

// 3.1 — monitor.py המרכזי
"""
monitor.py — WinAPI System Monitor
הרצת כל ה-collectors בthreads נפרדים
"""
import threading, signal, sys, time
import db
from collectors.windows   import run_window_tracker
from collectors.processes import run_process_scanner
from collectors.clipboard import run_clipboard_monitor
from collectors.sysinfo   import run_sysinfo

def banner():
    print("""
╔══════════════════════════════════════════╗
║  WinAPI System Monitor v1.0              ║
║  מגמת סייבר | תרגיל אסינכרוני 08        ║
╚══════════════════════════════════════════╝
  [Ctrl+C לעצירה מסודרת]
""")

def main():
    banner()
    db.init_db()

    # Event לעצירה מסודרת של כל ה-threads
    stop = threading.Event()

    # הגדרת threads
    threads = [
        threading.Thread(
            target=run_window_tracker,
            args=(2.0, stop),
            name="WindowTracker",
            daemon=True
        ),
        threading.Thread(
            target=run_process_scanner,
            args=(30.0, stop),
            name="ProcessScanner",
            daemon=True
        ),
        threading.Thread(
            target=run_clipboard_monitor,
            args=(3.0, stop),
            name="ClipboardMon",
            daemon=True
        ),
        threading.Thread(
            target=run_sysinfo,
            args=(60.0, stop),
            name="SysInfo",
            daemon=True
        ),
    ]

    # הפעל את כולם
    for t in threads:
        t.start()
        print(f"[+] {t.name} started")

    # טיפול ב-Ctrl+C
    def handle_exit(sig, frame):
        print("\n[!] Stopping all collectors...")
        stop.set()

    signal.signal(signal.SIGINT, handle_exit)
    signal.signal(signal.SIGTERM, handle_exit)

    # המתן לsignal עצירה
    while not stop.is_set():
        time.sleep(1)

    # המתן לסיום threads (מקסימום 5 שניות)
    for t in threads:
        t.join(timeout=5)
    print("[✓] Monitor stopped cleanly.")

if __name__ == "__main__":
    main()

// 4.1 — reporter.py + שאילתות SQL
"""reporter.py — ניתוח נתוני ה-monitor"""
import db
from datetime import datetime

def report_top_windows(limit=10):
    """TOP חלונות שביקרתי בהם"""
    print("\n═══ TOP ACTIVE WINDOWS ═══")
    conn = db.get_conn()
    rows = conn.execute("""
        SELECT window_title, process_name,
               COUNT(*) as visits
        FROM windows
        GROUP BY window_title
        ORDER BY visits DESC
        LIMIT ?
    """, (limit,)).fetchall()
    for r in rows:
        print(f"  {r['visits']:>4}x  [{r['process_name']}]  {r['window_title'][:55]}")
    conn.close()

def report_suspicious_clips():
    """clipboard entries עם תוכן רגיש"""
    print("\n═══ SENSITIVE CLIPBOARD EVENTS ═══")
    conn = db.get_conn()
    rows = conn.execute("""
        SELECT timestamp, content_type, content
        FROM clipboard
        WHERE content_type IN ('credential','credit_card','url')
        ORDER BY timestamp DESC
    """).fetchall()
    for r in rows:
        print(f"  {r['timestamp']}  [{r['content_type'].upper()}]"
              f"  {r['content'][:60]}")
    conn.close()

def report_new_processes():
    """תהליכים ב-C:\\Temp או %AppData% — חשוד!"""
    print("\n═══ SUSPICIOUS PROCESS PATHS ═══")
    conn = db.get_conn()
    rows = conn.execute("""
        SELECT DISTINCT name, path, MIN(timestamp) as first_seen
        FROM processes
        WHERE path LIKE '%\\Temp\\%'
           OR path LIKE '%\\AppData\\Local\\Temp\\%'
           OR path LIKE '%\\Downloads\\%'
        GROUP BY path
        ORDER BY first_seen DESC
    """).fetchall()
    if not rows:
        print("  לא נמצאו תהליכים חשודים ✓")
    for r in rows:
        print(f"  ⚠ {r['name']} — {r['path']}")
    conn.close()

def report_summary():
    """סיכום כללי"""
    print("\n═══ MONITORING SUMMARY ═══")
    conn = db.get_conn()
    for table in ["windows", "processes",
                   "clipboard", "system_snapshots"]:
        cnt = conn.execute(
            f"SELECT COUNT(*) FROM {table}").fetchone()[0]
        print(f"  {table:<22}: {cnt:>6} records")
    conn.close()

if __name__ == "__main__":
    report_summary()
    report_top_windows()
    report_suspicious_clips()
    report_new_processes()
