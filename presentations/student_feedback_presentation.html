def test_unknown_char(self):

מה חסר בטסטים שלכם

  • טסטי רובסטיות (None, non-string input)

  • טסטי ביצועים (נתונים גדולים)
  • טסטי גבולות (מילים ארוכות מאוד)

def reverse_words(word_list: list[str]) -> list[str]:
    """הופך את סדר האותיות בכל מילה ברשימה"""
    reversed_words = []
    for word in word_list:
        reversed_words.append(word[::-1])
    return reversed_words

    words = []

    for c in text:
        if c.isdigit():
            words.append(word[::-1])
