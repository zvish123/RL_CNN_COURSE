 session lifecycle

          
# Context Manager = פותח וסוגר אוטומטי
with Session(engine) as session:

    # 1. יצירת Python Object
    book = Book(title="הארי פוטר",
                price=74.99)

    # 2. הוסף ל-Session (טרם נשמר!)
    session.add(book)

    # 3. commit → כותב ל-DB
    session.commit()

    # 4. refresh → מושך חזרה (id!)
    session.refresh(book)
    return book

# ← session נסגר אוטומטית כאן

        
          
 dynamic filters

          
def get_books_by(in_stock=None, price=None):
    filters = []
    if in_stock is not None:
        filters.append(Book.in_stock == in_stock)
    if price is not None:
        filters.append(Book.price > price)

    stmt = select(Book).where(*filters)
    # *filters = unpacks the list!
    return session.execute(stmt).scalars().all()

💡 למה refresh?
אחרי commit, ה-id נוצר ב-DB. refresh שולף חזרה את הנתונים המעודכנים לאובייקט Python.
✅ scalars().all()
execute(stmt) מחזיר rows. scalars() מוציא את העמודה הראשונה (האובייקט עצמו). all() הופך לרשימה.

class Product(Base):
    __tablename__ = "products"
    id       = Column(Integer, ...)
    name     = Column(String(200))
    price    = Column(Float)
    in_stock = Column(Boolean)

    # ── @classmethod = שייך למחלקה ──
    @classmethod
    def add(cls, name, category, price):
        with Session(engine) as session:
            product = cls(name=name, ...)
            # cls = Product! גנרי!
            session.add(product)
            session.commit()
            session.refresh(product)
            return product

    @classmethod
    def get_all(cls):
        with Session(engine) as session:
            return session.execute(
                select(cls)).scalars().all()

    def delete(self):  # instance method
        with Session(engine) as session:
            obj = session.get(Product, self.id)
            session.delete(obj)
            session.commit()

class BaseRepo:
    """כל מחלקה שיורשת ממנה
    מקבלת CRUD חינם"""

    @classmethod
    def add(cls, **kwargs):
        with Session(engine) as session:
            obj = cls(**kwargs) # גנרי!
            session.add(obj)
            session.commit()
            session.refresh(obj)
            return obj

    @classmethod
    def delete_by_id(cls, id):
        with Session(engine) as session:
            obj = session.get(cls, id)
            if not obj:
                print("לא נמצא")
                return False
            session.delete(obj)
            session.commit()
            return True

    @classmethod
    def update_by_id(cls, id, **kwargs):
        with Session(engine) as session:
            obj = session.get(cls, id)
            if not obj: return None
            for key, value in kwargs.items():
                if hasattr(obj, key) and value is not None:
                    setattr(obj, key, value)
            session.commit()
            session.refresh(obj)
            return obj
🔑 הקסמים הטכניים
💡 **kwargs
cls(**kwargs) — מעביר כל ה-keyword arguments ישירות ל-constructor. לא צריך לדעת אילו שדות יש!
💡 setattr(obj, key, value)
setattr = עדכון דינמי של שדה לפי שם. במקום obj.price = value, כותבים setattr(obj, "price", value). מאפשר לולאה!
💡 hasattr(obj, key)
מוודא שהשדה קיים לפני העדכון — מגן מפני שגיאות אם מישהו מעביר שדה שלא קיים במודל.

class Customer(Base, BaseRepo):
    __tablename__ = "customers"

    id    = Column(Integer, primary_key=True)
    name  = Column(String(100), nullable=False)
    email = Column(String(150), unique=True)
    phone = Column(String(20))


class Order(Base, BaseRepo):
    __tablename__ = "orders"

    id          = Column(Integer, primary_key=True)

    # קשר ל-Customer
    customer_id = Column(Integer,
        ForeignKey("customers.id"))

    # קשר ל-Product
    product_id  = Column(Integer,
        ForeignKey("products.id"))

    quantity    = Column(Integer, default=1)
    ordered_at  = Column(DateTime,
        default=datetime.utcnow)

class Author(Base, BaseRepo):
    id    = Column(Integer, ...)
    name  = Column(String(100))
    birth_year = Column(Integer)

class Book(Base, BaseRepo):
    id        = Column(Integer, ...)
    title     = Column(String(200))
    author_id = Column(Integer,
        ForeignKey("authors.id"))
    pages     = Column(Integer)

class Loan(Base, BaseRepo):
    id        = Column(Integer, ...)
    book_id   = Column(Integer,
        ForeignKey("books.id"))
    member    = Column(String(100))
    due_date  = Column(DateTime)

def get_by_id(cls, id):
    with Session(engine) as session:
        return session.get(cls, id)

# הsession נסגר! הobject מנותק!

def add(cls, **kwargs):
    with Session(engine) as session:
        obj = cls(**kwargs)
        session.add(obj)
        session.commit()
        # שכחנו refresh!
        return obj

שאלון סיום

❓ מה היתרון המרכזי של class Product(Base, BaseRepo) על פני שלב ב׳?
הקוד רץ מהר יותר
לא צריך לכתוב מחדש CRUD לכל מודל — BaseRepo מספק הכל
אפשר להשתמש ב-ForeignKey
SQLite תומך ביותר סוגי נתונים
❓ מה עושה setattr(obj, "price", 149)?
שומר ישירות ל-DB
מוסיף שדה חדש לטבלה
מעדכן דינמית את obj.price = 149 — שקול לכתיבה ישירה אבל דרך שם השדה
מחזיר את ערך השדה הקיים
❓ אילו עקרונות SOLID מיושמים ב-base_repo + models?
רק OCP — Open/Closed Principle
רק ISP — Interface Segregation
SRP (כל קובץ אחראי אחד) + DRY (קוד אחת, שימוש רב)
LSP בלבד — Liskov Substitution
