    # קוד מיון איטי:
unsorted_list = [5, 3, 8, 6, 7, 2]
for i in range(len(unsorted_list)):
    for j in range(len(unsorted_list)-1):
        if unsorted_list[j] > unsorted_list[j+1]:
            unsorted_list[j], unsorted_list[j+1] = unsorted_list[j+1], unsorted_list[j]
print(unsorted_list)
    # פתרון יעיל:
unsorted_list = [5, 3, 8, 6, 7, 2]

def handle_client(client_socket):
    data = client_socket.recv(1024).decode()
    client_socket.send("נעניתי".encode())
    client_socket.close()

server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server_socket.bind(("localhost", 12345))
server_socket.listen(5)

while True:
    client_socket, addr = server_socket.accept()
    threading.Thread(target=handle_client, args=(client_socket,)).start()

    אתגר אינטגרציה: קוד קיסר מתמטי

    שלבו מתמטיקה (מודולו) עם קריפטוגרפיה:

    def caesar_encrypt(text, shift):
    result = ""
    for char in text:
        result += chr((ord(char) + shift) % 256)
    return result

print(caesar_encrypt("HELLO", 3))
    # פתרון משופר לאותיות באנגלית:
def caesar_encrypt(text, shift):
    result = ""
    for char in text:
        if char.isalpha():
            base = ord('A') if char.isupper() else ord('a')
            result += chr((ord(char) - base + shift) % 26 + base)
        else:
            result += char
    return result

print(caesar_encrypt("HELLO WORLD!", 3))  # KHOOR ZRUOG!

    # הוספת מעקב התקדמות:
import time

milestones = [("UI", 2), ("Logic", 3), ("Tests", 1)]
total_planned = sum(duration for _, duration in milestones)
completed = 0

for stage, duration in milestones:
    print(f"מתחילים {stage} - התקדמות: {completed}/{total_planned}")
    time.sleep(duration)
    completed += duration
    print(f"✅ הושלם {stage} - {(completed/total_planned)*100:.0f}%\n")

    פרויקט פתוח: מנתח טקסט

# הוסיפו עוד ניתוחים מעניינים!
    from collections import Counter

def analyze_text(text):
    words = text.lower().split()
    return {
        'word_count': len(words),
        'char_count': len(text),
        'most_common': Counter(words).most_common(3),
        'avg_word_length': sum(len(w) for w in words) / len(words),
        'unique_words': len(set(words))
    }

result = analyze_text("Hello world hello Python programming")
for key, value in result.items():
    print(f"{key}: {value}")

class ClassManager:
    def __init__(self):
        self.students = {}
        self.assignments = {}
        self.submissions = []
    
    def add_student(self, name, student_id):
        self.students[student_id] = {"name": name, "grades": []}
        print(f"✅ נוסף תלמיד: {name} (ID: {student_id})")
    
    def create_assignment(self, title, assignment_id, due_date):
        self.assignments[assignment_id] = {
            "title": title, 
            "due_date": due_date
        }
        print(f"📝 נוצרה משימה: {title}")
    
    def submit_assignment(self, student_id, assignment_id, grade):
        if student_id in self.students and assignment_id in self.assignments:
            self.students[student_id]["grades"].append(grade)
            print(f"📊 הוגשה משימה: ציון {grade}")
        else:
            print("❌ שגיאה: תלמיד או משימה לא קיימים")
    
    def get_class_average(self):
        all_grades = []
        for student in self.students.values():
            all_grades.extend(student["grades"])
        return sum(all_grades) / len(all_grades) if all_grades else 0
