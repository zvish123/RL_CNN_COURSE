רמות לוג

כל רמה מייצגת מידת חשיבות — חמור יותר = ערך גבוה יותר

0 DEBUG
מידע פיתוח — ערכים, מצבים פנימיים
log.debug("x = " + str(x))
1 INFO
אירועים רגילים — המערכת עובדת כשורה
log.info("שרת עלה בהצלחה")
2 WARNING
משהו חריג אבל המערכת ממשיכה
log.warning("קובץ חסר, משתמש בברירת מחדל")
3 ERROR
שגיאה — חלק מהמערכת נכשל
log.error("חיבור נכשל!")
💡 ב-Development: min_level=DEBUG  |  ב-Production: min_level=WARNING

הקוד — לב המנגנון

logger.py
import datetime DEBUG, INFO, WARNING, ERROR = 0, 1, 2, 3 LEVEL_NAMES = {DEBUG:'DEBUG', INFO:'INFO', WARNING:'WARNING', ERROR:'ERROR'} class Logger: def __init__(self, filepath='app.log', to_screen=True, to_file=True, min_level=DEBUG): self.filepath = filepath self.to_screen = to_screen self.to_file = to_file self.min_level = min_level if to_file: self._file = open(filepath, "a", encoding="utf-8") else: self._file = None

הקוד — _log + שימוש

_log + helpers
def _log(self, level, msg): if level < self.min_level: return ts = datetime.datetime.now() .strftime("%Y-%m-%d %H:%M:%S") ln = f'[{ts}][{LEVEL_NAMES[level]}] {msg}' if self.to_screen: print(ln) if self.to_file: self._file.write(ln + '\n') self._file.flush() def debug(self, m): self._log(DEBUG, m) def info(self, m): self._log(INFO, m) def warning(self, m): self._log(WARNING, m) def error(self, m): self._log(ERROR, m)

Context Manager — with

❌ ללא with
log = Logger('app.log') log.info('מתחיל') do_something() # 💀 שכחת? log.close() # קובץ נשאר פתוח!
✅ עם with
with Logger('app.log') as log: log.info('מתחיל') do_something() # ✅ סגור אוטומטית! # אפילו אם יש Exception
def __enter__(self): return self def __exit__(self, *args): self.close(); return False

Wrapper — הקוד המלא

logger_wrapper.py
import logging from logging.handlers import RotatingFileHandler class Logger: def __init__(self, name, filepath='app.log', to_screen=True, to_file=True, min_level=logging.DEBUG): lg = logging.getLogger(name) lg.setLevel(min_level) fmt = logging.Formatter( "%(asctime)s [%(levelname)s] %(message)s", datefmt="%Y-%m-%d %H:%M:%S") if to_file: fh = RotatingFileHandler( filepath, maxBytes=500_000, backupCount=2) fh.setFormatter(fmt); lg.addHandler(fh) if to_screen: sh = logging.StreamHandler() sh.setFormatter(fmt); lg.addHandler(sh) self._lg = lg
Public API — זהה לחלוטין!
def debug(self, m): self._lg.debug(m) def info(self, m): self._lg.info(m) def warning(self, m): self._lg.warning(m) def error(self, m): self._lg.error(m)

Decorator — @log_calls

דרך נוספת

במקום להוסיף log בכל פונקציה — עוטפים אותה פעם אחת

decorator.py
import functools, logging log = logging.getLogger("app") def log_calls(func): @functools.wraps(func) def wrapper(*args, **kwargs): log.info(f"▶ {func.__name__}() נקרא") try: result = func(*args, **kwargs) log.info(f"✓ {func.__name__}() הסתיים") return result except Exception as e: log.error(f"✗ {func.__name__}(): {e}") raise return wrapper
שימוש
@log_calls # ← שורה אחת! def load_data(filepath): # קוד פונקציה... return data @log_calls def connect_db(host): # ...
פלט אוטומטי
▶ load_data() נקרא ✓ load_data() הסתיים ▶ connect_db() נקרא ✗ connect_db(): Connection refused
🎯 יתרון: לא נוגעים בקוד הפנימי של הפונקציה — Separation of Concerns
כל פונקציה עם @log_calls מקבלת תיעוד מלא: כניסה, יציאה, שגיאות
