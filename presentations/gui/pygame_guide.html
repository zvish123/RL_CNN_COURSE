class Slider:
    def __init__(self, x, y, width, height=10,
                 min_value=0, max_value=100, initial_value=50,
                 label=""):
        """
        ××—×•×•×Ÿ ×”×–×–×”
        """
        self.x = x
        self.y = y
        self.width = width
        self.height = height
        
        self.min_value = min_value
        self.max_value = max_value
        self.value = initial_value
        self.label = label
        
        # ××™×“×•×ª
        self.track_rect = pygame.Rect(x, y, width, height)
        self.handle_radius = 12
        self.handle_x = self.value_to_position(initial_value)
        
        # ××¦×‘×™×
        self.dragging = False
        self.hovered = False
        
        # ×¦×‘×¢×™×
        self.track_color = (70, 70, 90)
        self.fill_color = (0, 255, 136)
        self.handle_color = (255, 255, 255)
        self.hover_color = (0, 212, 255)
        self.text_color = (255, 255, 255)
        
        self.font = pygame.font.SysFont('Arial', 20)
    
    def value_to_position(self, value):
        """
        ×”××¨×ª ×¢×¨×š ×œ××™×§×•× ×¢×œ ×”××¡×š
        """
        percentage = (value - self.min_value) / (self.max_value - self.min_value)
        return self.x + int(percentage * self.width)
    
    def position_to_value(self, pos_x):
        """
        ×”××¨×ª ××™×§×•× ×œ×¢×¨×š
        """
        # ×”×’×‘×œ×” ×œ×’×‘×•×œ×•×ª
        pos_x = max(self.x, min(pos_x, self.x + self.width))
        percentage = (pos_x - self.x) / self.width
        value = self.min_value + percentage * (self.max_value - self.min_value)
        return round(value, 2)
    
    def handle_event(self, event):
        """
        ×˜×™×¤×•×œ ×‘××™×¨×•×¢×™×
        """
        if event.type == pygame.MOUSEMOTION:
            # ×‘×“×™×§×” ×× ×¢×œ ×”×™×“×™×ª
            dx = event.pos[0] - self.handle_x
            dy = event.pos[1] - self.track_rect.centery
            distance = (dx**2 + dy**2) ** 0.5
            self.hovered = distance <= self.handle_radius + 5
            
            # ×’×¨×™×¨×”
            if self.dragging:
                self.handle_x = event.pos[0]
                self.value = self.position_to_value(self.handle_x)
                self.handle_x = self.value_to_position(self.value)
                return True
        
        if event.type == pygame.MOUSEBUTTONDOWN:
            if event.button == 1:
                # ×‘×“×™×§×” ×× ×œ×—×¦×• ×¢×œ ×”×™×“×™×ª
                dx = event.pos[0] - self.handle_x
                dy = event.pos[1] - self.track_rect.centery
                distance = (dx**2 + dy**2) ** 0.5
                
                if distance <= self.handle_radius + 5:
                    self.dragging = True
                # ××• ×œ×—×™×¦×” ×¢×œ ×”××¡×™×œ×” ×¢×¦××”
                elif self.track_rect.collidepoint(event.pos):
                    self.handle_x = event.pos[0]
                    self.value = self.position_to_value(self.handle_x)
                    self.handle_x = self.value_to_position(self.value)
                    self.dragging = True
                    return True
        
        if event.type == pygame.MOUSEBUTTONUP:
            if event.button == 1:
                self.dragging = False
        
        return False
    
    def draw(self, screen):
        """
        ×¦×™×•×¨ ×”××—×•×•×Ÿ
        """
        # ×˜×§×¡×˜ ×ª×•×•×™×ª
        if self.label:
            label_surface = self.font.render(self.label, True, self.text_color)
            screen.blit(label_surface, (self.x, self.y - 30))
        
        # ××¡×™×œ×” ×¨×§×¢
        pygame.draw.rect(screen, self.track_color, self.track_rect, border_radius=5)
        
        # ××¡×™×œ×” ×××•×œ××ª (×¢×“ ×”×™×“×™×ª)
        fill_width = self.handle_x - self.x
        if fill_width > 0:
            fill_rect = pygame.Rect(self.x, self.y, fill_width, self.height)
            pygame.draw.rect(screen, self.fill_color, fill_rect, border_radius=5)
        
        # ×™×“×™×ª
        handle_color = self.hover_color if (self.hovered or self.dragging) else self.handle_color
        handle_pos = (self.handle_x, self.track_rect.centery)
        
        # ×¦×œ ×œ×™×“×™×ª
        shadow_pos = (handle_pos[0] + 2, handle_pos[1] + 2)

        # ×”×™×“×™×ª ×¢×¦××”

        value_rect = value_surface.get_rect(center=(handle_pos[0], handle_pos[1] - 25))

class IconButton:
    def __init__(self, x, y, icon_path, size=64, 
                 bg_color=(70, 70, 90),
                 hover_color=(100, 100, 130)):
        """
        ×›×¤×ª×•×¨ ×¢× ××™×™×§×•×Ÿ
        """
        self.rect = pygame.Rect(x, y, size, size)
        
        # ×˜×¢×™× ×ª ××™×™×§×•×Ÿ

        self.bg_color = bg_color
        self.hover_color = hover_color
        self.hovered = False
        self.pressed = False
    
    def handle_event(self, event):
        if event.type == pygame.MOUSEMOTION:
            self.hovered = self.rect.collidepoint(event.pos)
        
        if event.type == pygame.MOUSEBUTTONDOWN:
            if event.button == 1 and self.hovered:
                self.pressed = True
        
        if event.type == pygame.MOUSEBUTTONUP:
            if event.button == 1 and self.pressed:
                self.pressed = False
                if self.hovered:
                    return True
        
        return False
    
    def draw(self, screen):
        # ×¨×§×¢
        color = self.hover_color if self.hovered else self.bg_color
        rect = self.rect.copy()
        if self.pressed:
            rect.y += 2
        
        pygame.draw.rect(screen, color, rect, border_radius=10)
        
        # ××¡×’×¨×ª

import pygame
import random
from screens.base_screen import BaseScreen
from settings import *

class SnakeGameScreen(BaseScreen):
    def __init__(self, manager):
        super().__init__(manager)
        
        # ×”×’×“×¨×•×ª ××©×—×§
        self.cell_size = 20
        self.grid_width = SCREEN_WIDTH // self.cell_size
        self.grid_height = SCREEN_HEIGHT // self.cell_size
        
        # × ×—×©
        self.snake = [(self.grid_width // 2, self.grid_height // 2)]
        self.direction = (1, 0)  # ×™××™× ×”
        self.next_direction = (1, 0)
        
        # ××•×›×œ
        self.food = self.spawn_food()
        
        # ××©×—×§
        self.score = 0
        self.game_over = False
        self.move_timer = 0
        self.move_delay = 0.1  # ×ª× ×•×¢×” ×›×œ 0.1 ×©× ×™×•×ª
        
        # ×¤×•× ×˜
        self.font = pygame.font.SysFont('Arial', 30)
    
    def spawn_food(self):
        while True:
            food = (
                random.randint(0, self.grid_width - 1),
                random.randint(0, self.grid_height - 1)
            )
            if food not in self.snake:
                return food
    
    def handle_event(self, event):
        if event.type == pygame.KEYDOWN and not self.game_over:
            # ×©×™× ×•×™ ×›×™×•×•×Ÿ (××™ ××¤×©×¨ ×œ×”×ª×”×¤×š 180 ××¢×œ×•×ª)
            if event.key == pygame.K_UP and self.direction != (0, 1):
                self.next_direction = (0, -1)
            elif event.key == pygame.K_DOWN and self.direction != (0, -1):
                self.next_direction = (0, 1)
            elif event.key == pygame.K_LEFT and self.direction != (1, 0):
                self.next_direction = (-1, 0)
            elif event.key == pygame.K_RIGHT and self.direction != (-1, 0):
                self.next_direction = (1, 0)
        
        if event.type == pygame.KEYDOWN and self.game_over:
            if event.key == pygame.K_SPACE:
                self.__init__(self.manager)  # ×”×ª×—×œ ××—×“×©
            elif event.key == pygame.K_ESCAPE:
                from screens.menu_screen import MenuScreen
                self.manager.change_screen(MenuScreen(self.manager, "Player"))
    
    def update(self, dt):
        if self.game_over:
            return
        
        self.move_timer += dt
        
        if self.move_timer >= self.move_delay:
            self.move_timer = 0
            self.direction = self.next_direction
            
            # ×ª× ×•×¢×ª ×”× ×—×©
            head = self.snake[0]
            new_head = (
                head[0] + self.direction[0],
                head[1] + self.direction[1]
            )
            
            # ×‘×“×™×§×ª ×”×ª× ×’×©×•×ª ×‘×§×™×¨
            if (new_head[0] < 0 or new_head[0] >= self.grid_width or
                new_head[1] < 0 or new_head[1] >= self.grid_height):
                self.game_over = True
                return
            
            # ×‘×“×™×§×ª ×”×ª× ×’×©×•×ª ×‘×’×•×£
            if new_head in self.snake:
                self.game_over = True
                return
            
            # ×”×•×¡×¤×ª ×¨××© ×—×“×©
            self.snake.insert(0, new_head)
            
            # ×‘×“×™×§×” ×× ××›×œ
            if new_head == self.food:
                self.score += 10
                self.food = self.spawn_food()
                # ×”× ×—×© ×’×“×œ - ×œ× ××•×¦×™××™× ××ª ×”×–× ×‘
            else:
                # ×”× ×—×© ×œ× ×’×“×œ - ××•×¦×™××™× ××ª ×”×–× ×‘
                self.snake.pop()
    
    def draw(self, screen):
        screen.fill((20, 20, 30))
        
        # ×¨×©×ª
        for x in range(0, SCREEN_WIDTH, self.cell_size):
            pygame.draw.line(screen, (40, 40, 50), (x, 0), (x, SCREEN_HEIGHT))
        for y in range(0, SCREEN_HEIGHT, self.cell_size):
            pygame.draw.line(screen, (40, 40, 50), (0, y), (SCREEN_WIDTH, y))
        
        # ××•×›×œ
        food_rect = pygame.Rect(
            self.food[0] * self.cell_size,
            self.food[1] * self.cell_size,
            self.cell_size - 2,
            self.cell_size - 2
        )
        pygame.draw.rect(screen, (255, 0, 0), food_rect, border_radius=5)
        
        # × ×—×©
        for i, segment in enumerate(self.snake):
            color = (0, 255, 100) if i == 0 else (0, 200, 80)
            segment_rect = pygame.Rect(
                segment[0] * self.cell_size,
                segment[1] * self.cell_size,
                self.cell_size - 2,
                self.cell_size - 2
            )
            pygame.draw.rect(screen, color, segment_rect, border_radius=3)
        
        # × ×™×§×•×“
        score_text = self.font.render(f'Score: {self.score}', True, (255, 255, 255))
        screen.blit(score_text, (10, 10))
        
        # Game Over
        if self.game_over:
            overlay = pygame.Surface((SCREEN_WIDTH, SCREEN_HEIGHT))
            overlay.set_alpha(200)
            overlay.fill((0, 0, 0))
            screen.blit(overlay, (0, 0))

            game_over_text = game_over_font.render('GAME OVER', True, (255, 0, 0))

            restart_text = self.font.render('Press SPACE to restart', True, (200, 200, 200))
            menu_text = self.font.render('Press ESC for menu', True, (200, 200, 200))

                       (SCREEN_WIDTH // 2 - game_over_text.get_width() // 2, 200))

import json
import os

class HighScoreManager:
    def __init__(self, filename='highscores.json'):
        self.filename = filename
        self.scores = self.load_scores()
    
    def load_scores(self):
        if os.path.exists(self.filename):
            with open(self.filename, 'r') as f:
                return json.load(f)
        return []
    
    def save_scores(self):
        with open(self.filename, 'w') as f:
            json.dump(self.scores, f, indent=2)
    
    def add_score(self, name, score):
        self.scores.append({'name': name, 'score': score})
        # ××™×•×Ÿ ×œ×¤×™ × ×™×§×•×“ (×”×›×™ ×’×‘×•×” ×¨××©×•×Ÿ)
        self.scores.sort(key=lambda x: x['score'], reverse=True)

        self.scores = self.scores[:10]
        self.save_scores()
    
    def get_top_scores(self, count=10):
        return self.scores[:count]
    
    def is_high_score(self, score):
        if len(self.scores) < 10:
            return True
        return score > self.scores[-1]['score']
