04 — CREATE — הוספת שורות

CREATE — הוספת נתונים

יצירת instance → הוספה ל-Session → commit לדיסק

הוספת ספר אחד

with Session(engine) as session:

    # 1. צור Python object
    book = Book(
        title="הארי פוטר",
        author="J.K. Rowling",
        price=79.90,
        in_stock=True
    )

    # 2. הוסף ל-Session (עוד לא נשמר!)
    session.add(book)

    # 3. commit → כתב ל-DB
    session.commit()

    # id נוצר אוטומטית אחרי commit
    print(f"נשמר עם id={book.id}")

from sqlalchemy import select

with Session(engine) as session:

    # כל הספרים
    stmt = select(Book)
    all_books = session.execute(stmt).scalars().all()

שאילתות מורכבות

from sqlalchemy import select, func, or_, and_

with Session(engine) as session:

    # ספירה
    count = session.execute(
        select(func.count(Book.id))
    ).scalar()

    # ממוצע מחיר
    avg = session.execute(
        select(func.avg(Book.price))
    ).scalar()

    # or_ — OR בתנאי
    stmt = select(Book).where(
        or_(
            Book.author == "Orwell",
            Book.price < 50
        )
    )

    # LIKE — חיפוש חלקי
    stmt = select(Book).where(
        Book.title.like("%פוטר%")
    )

scalar() vs. all()

06 — UPDATE — עדכון שורות

UPDATE — עדכון נתונים

שלוף object, שנה שדות, commit — SQLAlchemy עוקב אחרי שינויים

עדכון object בודד

with Session(engine) as session:

    # שלב 1: שלוף את ה-object
    book = session.get(Book, 1)

    if book:
        # שלב 2: שנה שדות
        book.price = 89.90
        book.in_stock = False

        # שלב 3: commit — SQLAlchemy
        # מזהה שהשתנה ומייצר UPDATE
        session.commit()
        print("✓ עודכן בהצלחה")
Unit of Work: SQLAlchemy עוקב אחרי שינויים ב-objects. כשמבצעים commit() — רק השדות שהשתנו נכתבים ל-DB.

עדכון מסה — update()

from sqlalchemy import update

with Session(engine) as session:
    # עדכן כל ספרי Orwell — מחיר+10
    stmt = (
        update(Book)
        .where(Book.author == "Orwell")
        .values(price=Book.price + 10)
    )
    session.execute(stmt)
    session.commit()

מה SQLAlchemy מייצר?

# book.price = 89.90 → מייצר:
# UPDATE books
# SET price = 89.90, in_stock = 0
# WHERE books.id = 1

# bulk update → מייצר:
# UPDATE books
# SET price = price + 10
# WHERE books.author = 'Orwell'

עדכון בטוח — בדוק לפני

def update_price(session, book_id, new_price):
    if new_price < 0:
        raise ValueError("מחיר לא יכול להיות שלילי")

    book = session.get(Book, book_id)
    if not book:
        raise ValueError(f"ספר {book_id} לא נמצא")

    old = book.price
    book.price = new_price
    session.commit()
    return f"עודכן: {old}→{new_price}"
💡 rollback(): אם משהו נכשל — session.rollback() מבטל את כל השינויים מה-commit האחרון.
07 — DELETE — מחיקת שורות

DELETE — מחיקת נתונים

session.delete(obj) למחיקת object, delete().where() למחיקת מסה

מחיקת שורה אחת

with Session(engine) as session:

    # שלב 1: שלוף
    book = session.get(Book, 3)

    if book:
        # שלב 2: מחק
        session.delete(book)

        # שלב 3: commit
        session.commit()
        print(f"נמחק: {book.title}")
    else:
        print("ספר לא נמצא")

מחיקת מסה — delete().where()

from sqlalchemy import delete

with Session(engine) as session:

    # מחק ספרים שאינם במלאי
    stmt = (
        delete(Book)
        .where(Book.in_stock == False)
    )
    result = session.execute(stmt)
    session.commit()

    print(f"נמחקו: {result.rowcount} שורות")

מחיקה עם אישור — פונקציה בטוחה

def safe_delete(session, book_id):
    """מחיקה בטוחה עם בדיקות"""
    book = session.get(Book, book_id)
    if not book:
        return False, "לא נמצא"

    title = book.title
    try:
        session.delete(book)
        session.commit()
        return True, f"נמחק: {title}"
    except Exception as e:
        session.rollback()
        return False, f"שגיאה: {e}"
⚠️ מחיקה היא בלתי הפיכה! לאחר commit, הנתונים אבודים. בשיטות ממשיות נהוג להשתמש ב-"Soft Delete" — שדה is_deleted=True במקום מחיקה אמיתית.

Soft Delete — מחיקה רכה

# עדיף! שמור את הנתונים
class Book(Base):
    ...
    is_deleted = Column(Boolean,
                         default=False)

# "מחיקה"
book.is_deleted = True
session.commit()

# שליפה רק של "קיימים"
stmt = select(Book).where(
    Book.is_deleted == False
)

מודל Product + כל פעולות CRUD בפונקציות נפרדות — לא קשורה לתרגיל הצ'אט
# product_store.py — דוגמה מלאה
from sqlalchemy import (
    create_engine, Column,
    Integer, String, Float, Boolean
)
from sqlalchemy.orm import (
    declarative_base, Session
)
from sqlalchemy import select, update, delete

engine = create_engine("sqlite:///store.db")
Base   = declarative_base()

class Product(Base):
    __tablename__ = "products"
    id       = Column(Integer, primary_key=True)
    name     = Column(String(100), nullable=False)
    price    = Column(Float)
    category = Column(String(50))
    active   = Column(Boolean, default=True)

    def __repr__(self):
        return f"<Product {self.id}: {self.name}>"

Base.metadata.create_all(engine)

# ─── CREATE ───────────────────────────
def add_product(name, price, category):
    with Session(engine) as s:
        p = Product(name=name,
                    price=price,
                    category=category)
        s.add(p)
        s.commit()
        return p.id
      
# ─── READ ────────────────────────────
def get_by_category(category):
    with Session(engine) as s:
        stmt = select(Product).where(
            Product.category == category,
            Product.active == True
        ).order_by(Product.price)
        return s.execute(stmt).scalars().all()

# ─── UPDATE ──────────────────────────
def apply_discount(category, pct):
    """הנחה באחוזים לקטגוריה"""
    with Session(engine) as s:
        stmt = (
            update(Product)
            .where(Product.category == category)
            .values(
                price=Product.price * (1 - pct/100)
            )
        )
        s.execute(stmt)
        s.commit()

# ─── DELETE (soft) ───────────────────
def deactivate(product_id):
    """מחיקה רכה — לא מוחקים ממש"""
    with Session(engine) as s:
        p = s.get(Product, product_id)
        if p:
            p.active = False
            s.commit()

# ─── שימוש ───────────────────────────
add_product("מחשב נייד", 3500, "אלקטרוניקה")
add_product("אוזניות",   249,  "אלקטרוניקה")
apply_discount("אלקטרוניקה", 10)  # -10%
items = get_by_category("אלקטרוניקה")

def register_user(username, password):
    with Session(engine) as s:
        if s.get(User, username):
            return False, "קיים כבר"
        s.add(User(username=username,
                    password=password))
        s.commit()
        return True, "נרשם"

def authenticate(username, password):
    with Session(engine) as s:
        u = s.get(User, username)
        return u is not None and u.password == password
